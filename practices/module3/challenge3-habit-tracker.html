Habit Tracker

Difficulty: Medium Module: Local Storage Concepts: Hive, Streak Tracking, Date Logic Est. Time: 35-45 min 📦 Hive

📋 Challenge Description

Build a Habit Tracker app that helps users track their daily habits and maintain streaks. All data should be stored locally using Hive as the local database.

✅ Requirements

  • Create a Habit model with id, name, icon, color, createdAt
  • Create a HabitLog model to track daily completion
  • Use Hive for local storage with custom adapters
  • Use Riverpod for state management
  • Display a dashboard showing all habits with:
    • Current streak count
    • Today's completion status
    • Weekly progress indicator
  • Allow users to toggle completion for today
  • Show a history view (last 7 days) with visual indicators
  • Calculate streaks correctly (consecutive days)
  • Handle empty state when no habits exist

📦 Hive Setup

Hive is a lightweight, fast key-value database for Flutter.

flutter pub add hive_flutter hive_generator build_runner
💡 Note: Use type adapters for both Habit and HabitLog models.

📚 Resources

  • Hive Package
  • Hive Flutter
  • Hive Documentation
  • Riverpod Documentation

💡 Hint: Use two Hive boxes: one for habits and one for habit logs. For streak calculation, check if the habit was completed yesterday, then count backwards. Use DateTime.now().difference() to check dates. Create a HabitLog with habitId, date, and isCompleted.

5. Habit Provider (lib/providers/habit_provider.dart)

lib/providers/habit_provider.dart import 'package:flutter_riverpod/flutter_riverpod.dart'; import 'package:hive_flutter/hive_flutter.dart'; import '../models/habit.dart'; import '../models/habit_log.dart'; final habitBoxProvider = Provider<Box<Habit>>((ref) { return Hive.box<Habit>('habits'); }); final habitLogBoxProvider = Provider<Box<HabitLog>>((ref) { return Hive.box<HabitLog>('habit_logs'); }); class HabitNotifier extends StateNotifier<List<Habit>> { final Box<Habit> _habitBox; final Box<HabitLog> _logBox; HabitNotifier(this._habitBox, this._logBox) : super(const []) { _loadHabits(); } void _loadHabits() { state = _habitBox.values.toList() ..sort((a, b) => a.createdAt.compareTo(b.createdAt)); } void addHabit(String name, String icon, int colorValue) { final habit = Habit.create( name: name.trim(), icon: icon, colorValue: colorValue, ); _habitBox.add(habit); _loadHabits(); } void deleteHabit(Habit habit) { final index = _habitBox.values.toList().indexOf(habit); if (index != -1) { // Delete all logs for this habit final logsToDelete = _logBox.values .where((log) => log.habitId == habit.id) .toList(); for (var log in logsToDelete) { final logIndex = _logBox.values.toList().indexOf(log); if (logIndex != -1) { _logBox.deleteAt(logIndex); } } _habitBox.deleteAt(index); _loadHabits(); } } void toggleHabit(String habitId) { final today = DateTime.now(); final existingLog = _logBox.values .firstWhere( (log) => log.habitId == habitId && _isSameDay(log.date, today), orElse: () => throw Exception('Log not found'), ); final index = _logBox.values.toList().indexOf(existingLog); if (index != -1) { final updatedLog = existingLog.copyWith( isCompleted: !existingLog.isCompleted, ); _logBox.putAt(index, updatedLog); } } bool _isSameDay(DateTime a, DateTime b) { return a.year == b.year && a.month == b.month && a.day == b.day; } int getStreak(String habitId) { final logs = _logBox.values .where((log) => log.habitId == habitId) .toList() ..sort((a, b) => b.date.compareTo(a.date)); if (logs.isEmpty) return 0; int streak = 0; DateTime currentDate = DateTime.now(); for (var log in logs) { if (!_isSameDay(log.date, currentDate) && currentDate.difference(log.date).inDays > 1) { break; } if (log.isCompleted) { streak++; currentDate = log.date; } else { break; } } return streak; } bool isCompletedToday(String habitId) { final today = DateTime.now(); final log = _logBox.values.firstWhere( (log) => log.habitId == habitId && _isSameDay(log.date, today), orElse: () => HabitLog.incomplete(habitId: habitId), ); return log.isCompleted; } bool wasCompletedOn(String habitId, DateTime date) { final log = _logBox.values.firstWhere( (log) => log.habitId == habitId && _isSameDay(log.date, date), orElse: () => HabitLog.incomplete(habitId: habitId), ); return log.isCompleted; } void ensureTodayLog(String habitId) { final today = DateTime.now(); final exists = _logBox.values.any( (log) => log.habitId == habitId && _isSameDay(log.date, today), ); if (!exists) { _logBox.add( HabitLog.incomplete(habitId: habitId), ); } } } final habitProvider = StateNotifierProvider<HabitNotifier, List<Habit>>((ref) { final habitBox = ref.watch(habitBoxProvider); final logBox = ref.watch(habitLogBoxProvider); return HabitNotifier(habitBox, logBox); });
