Expense Tracker Pro

Difficulty: Medium Module: Local Storage Concepts: SQLite, CRUD, Query Filtering, Charts Est. Time: 40-50 min 🗄️ SQLite

📋 Challenge Description

Build an Expense Tracker Pro app that allows users to track their income and expenses. All data should be stored locally using SQLite with the sqflite package.

✅ Requirements

  • Create an Expense model with id , title , amount , category , type (income/expense), date
  • Implement CRUD operations using SQLite
  • Use sqflite package for database operations
  • Use Riverpod for state management
  • Show a summary dashboard with:
    • Total income
    • Total expenses
    • Net balance
  • Display a list of transactions sorted by date
  • Add filter by category functionality
  • Add filter by date range (optional)
  • Include a chart showing spending by category
  • Handle empty state when no transactions exist

🗄️ SQLite Setup

sqflite is a SQLite plugin for Flutter with full SQL support.

flutter pub add sqflite path_provider
💡 Note: SQLite provides powerful querying capabilities for filtering and aggregating data.

📚 Resources

  • sqflite Package
  • path_provider
  • Flutter SQLite Cookbook
  • fl_chart for charts

💡 Hint: Create a database helper class with methods for insert , update , delete , and query . Use FutureProvider for the list of transactions and summary data. For charts, use fl_chart package to display a pie chart of spending by category.

class Expense {
  final int? id;

  final String title;

  final double amount;

  final String category;

  final String type;

  // 'income' or 'expense'

  final DateTime date;

  Expense({
    this.id,

    required this.title,

    required this.amount,

    required this.category,

    required this.type,

    required this.date,
  });

  Map<String, dynamic> toMap() {
    return {
      'id': id,

      'title': title,

      'amount': amount,

      'category': category,

      'type': type,

      'date': date.toIso8601String(),
    };
  }

  factory Expense.fromMap(Map<String, dynamic> map) {
    return Expense(
      id: map['id'],
      title: map['title'],
      amount: map['amount'],
      category: map['category'],
      type: map['type'],
      date: DateTime.parse(map['date']),
    );
  }

  String getFormattedDate() {
    return DateFormat.yMMMd().add_jm().format(date);
  }

  static List<String> get categories => [
    'Food & Dining',

    'Transportation',

    'Shopping',

    'Entertainment',

    'Bills & Utilities',

    'Health & Fitness',

    'Education',

    'Other',
  ];

  static List<String> get incomeCategories => [
    'Salary',

    'Freelance',

    'Investment',

    'Gift',

    'Other',
  ];

  static List<String> get expenseCategories => [
    'Food & Dining',

    'Transportation',

    'Shopping',

    'Entertainment',

    'Bills & Utilities',

    'Health & Fitness',

    'Education',

    'Other',
  ];
}

3. Database Helper (lib/database/database_helper.dart)

lib/database/database_helper.dart 
import 'package:sqflite/sqflite.dart';

import 'package:path/path.dart';

import '../models/expense.dart';

class DatabaseHelper {
  static final DatabaseHelper _instance = DatabaseHelper._internal();

  static Database? _database;

  DatabaseHelper._internal();

  factory DatabaseHelper() => _instance;

  Future<Database> get database async {
    if (_database != null) return _database!;
    _database = await _initDatabase();

    return _database!;
  }

  Future<Database> _initDatabase() async {
    String path = await getDatabasesPath();
    String dbPath = join(path, 'expenses.db');

    return await openDatabase(dbPath, version: 1, onCreate: _onCreate);
  }

  Future<void> _onCreate(Database db, int version) async {
    await db.execute('''
        CREATE TABLE expenses(
          id INTEGER PRIMARY KEY AUTOINCREMENT,
          title TEXT NOT NULL,
          amount REAL NOT NULL,
          category TEXT NOT NULL,
          type TEXT NOT NULL,
          date TEXT NOT NULL
        )
      ''');
  }

  // CRUD Operations

  Future<int> insertExpense(Expense expense) async {
    Database db = await this.database;

    return await db.insert(
      'expenses',
      expense.toMap(),
      conflictAlgorithm: ConflictAlgorithm.replace,
    );
  }

  Future<List<Expense>> getAllExpenses() async {
    Database db = await this.database;

    final List<Map<String, dynamic>> maps = await db.query(
      'expenses',
      orderBy: 'date DESC',
    );

    return List.generate(maps.length, (i) => Expense.fromMap(maps[i]));
  }

  Future<List<Expense>> getExpensesByCategory(String category) async {
    Database db = await this.database;

    final List<Map<String, dynamic>> maps = await db.query(
      'expenses',
      where: 'category = ?',
      whereArgs: [category],
      orderBy: 'date DESC',
    );

    return List.generate(maps.length, (i) => Expense.fromMap(maps[i]));
  }

  Future<List<Expense>> getExpensesByType(String type) async {
    Database db = await this.database;

    final List<Map<String, dynamic>> maps = await db.query(
      'expenses',
      where: 'type = ?',
      whereArgs: [type],
      orderBy: 'date DESC',
    );

    return List.generate(maps.length, (i) => Expense.fromMap(maps[i]));
  }

  Future<Map<String, double>> getCategorySpending() async {
    Database db = await this.database;

    final List<Map<String, dynamic>> results = await db.rawQuery('''
        SELECT category, SUM(amount) as total
        FROM expenses
        WHERE type = 'expense'
        GROUP BY category
        ORDER BY total DESC
      ''');

    Map<String, double> spending = {};

    for (var row in results) {
      spending[row['category']] = row['total'];
    }

    return spending;
  }

  Future<Map<String, double>> getSummary() async {
    Database db = await this.database;

    final List<Map<String, dynamic>> results = await db.rawQuery('''
        SELECT
          SUM(CASE WHEN type = 'income' THEN amount ELSE 0 END) as total_income,
          SUM(CASE WHEN type = 'expense' THEN amount ELSE 0 END) as total_expense
        FROM expenses
      ''');

    return {
      'total_income': results[0]['total_income'] ?? 0.0,

      'total_expense': results[0]['total_expense'] ?? 0.0,
    };
  }

  Future<int> updateExpense(Expense expense) async {
    Database db = await this.database;

    return await db.update(
      'expenses',
      expense.toMap(),
      where: 'id = ?',
      whereArgs: [expense.id],
    );
  }

  Future<int> deleteExpense(int id) async {
    Database db = await this.database;

    return await db.delete('expenses', where: 'id = ?', whereArgs: [id]);
  }
}

4. Expense Providers (lib/providers/expense_provider.dart)

lib/providers/expense_provider.dart 
import 'package:flutter_riverpod/flutter_riverpod.dart';

import '../models/expense.dart';

import '../database/database_helper.dart';

final dbHelper = DatabaseHelper();

// State for filtering

final filterCategoryProvider = StateProvider<String?>((ref) => null);

final filterTypeProvider = StateProvider<String?>((ref) => null);

final expenseListProvider = FutureProvider<List<Expense>>((ref) async {
  final category = ref.watch(filterCategoryProvider);

  final type = ref.watch(filterTypeProvider);

  if (category != null) {
    return await dbHelper.getExpensesByCategory(category);
  }

  if (type != null) {
    return await dbHelper.getExpensesByType(type);
  }

  return await dbHelper.getAllExpenses();
});

final summaryProvider = FutureProvider<Map<String, double>>((ref) async {
  return await dbHelper.getSummary();
});

final categorySpendingProvider = FutureProvider<Map<String, double>>((
  ref,
) async {
  return await dbHelper.getCategorySpending();
});

class ExpenseNotifier extends StateNotifier<void> {
  final Ref ref;

  ExpenseNotifier(this.ref) : super(null);

  Future<void> addExpense(Expense expense) async {
    await dbHelper.insertExpense(expense);
    ref.invalidate(expenseListProvider);
    ref.invalidate(summaryProvider);
    ref.invalidate(categorySpendingProvider);
  }

  Future<void> updateExpense(Expense expense) async {
    await dbHelper.updateExpense(expense);
    ref.invalidate(expenseListProvider);
    ref.invalidate(summaryProvider);
    ref.invalidate(categorySpendingProvider);
  }

  Future<void> deleteExpense(int id) async {
    await dbHelper.deleteExpense(id);
    ref.invalidate(expenseListProvider);
    ref.invalidate(summaryProvider);
    ref.invalidate(categorySpendingProvider);
  }

  void clearFilters() {
    ref.read(filterCategoryProvider.notifier).state = null;
    ref.read(filterTypeProvider.notifier).state = null;
    ref.invalidate(expenseListProvider);
  }

  void setCategoryFilter(String? category) {
    ref.read(filterCategoryProvider.notifier).state = category;
    ref.read(filterTypeProvider.notifier).state = null;
    ref.invalidate(expenseListProvider);
  }

  void setTypeFilter(String? type) {
    ref.read(filterTypeProvider.notifier).state = type;
    ref.read(filterCategoryProvider.notifier).state = null;
    ref.invalidate(expenseListProvider);
  }
}

final expenseNotifierProvider = StateNotifierProvider<ExpenseNotifier, void>((
  ref,
) {
  return ExpenseNotifier(ref);
});

  }

  void _saveExpense(BuildContext context) {
    if (!_formKey.currentState()!.validate()) return;

    final expense = Expense(
      id: isEditing ? widget.expense!.id : null,
      title: _titleController.text.trim(),
      amount: double.parse(_amountController.text.trim()),
      category: _selectedCategory,
      type: _selectedType,
      date: _selectedDate,
    );

    setState(() => _isLoading = true);

    Future.delayed(Duration(milliseconds: 500), () {
      final notifier = ref.read(expenseNotifierProvider.notifier);

      if (isEditing) {
        notifier.updateExpense(expense);
      } else {
        notifier.addExpense(expense);
      }
      Navigator.pop(context);
    });
  }
}

9. Category Chart Widget (lib/screens/widgets/category_chart.dart)

import 'package:flutter/material.dart';

import 'package:fl_chart/fl_chart.dart';

import 'dart:math';

class CategoryChart extends StatelessWidget {
  final Map<String, double> spending;

  const CategoryChart({required this.spending, super.key});

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: EdgeInsets.all(16),
      child: Card(
        elevation: 4,
        child: Padding(
          padding: EdgeInsets.all(16),
          child: Column(
            children: [
              Text(
                'Spending by Category',
                style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
              ),
              SizedBox(height: 16),
              SizedBox(height: 250, child: _buildPieChart()),
              SizedBox(height: 16),
              _buildLegend(),
            ],
          ),
        ),
      ),
    );
  }

  Widget _buildPieChart() {
    final colors = [
      Colors.blue,
      Colors.red,
      Colors.green,
      Colors.orange,
      Colors.purple,
      Colors.pink,
      Colors.teal,
      Colors.amber,
    ];

    final entries = spending.entries.toList();

    final total = entries.fold(0.0, (sum, e) => sum + e.value);

    return PieChart(
      PieChartData(
        sections: entries.asMap().entries.map((entry) {
          final index = entry.key;

          final name = entry.value.key;

          final amount = entry.value.value;

          final percentage = total > 0 ? (amount / total) * 100 : 0.0;

          return PieChartSectionData(
            color: colors[index % colors.length],
            value: amount,
            title: percentage > 5 ? '\${amount.toStringAsFixed(0)}' : '',
            radius: 100,
            titleStyle: TextStyle(
              fontSize: 12,
              fontWeight: FontWeight.bold,
              color: Colors.white,
            ),
          );
        }).toList(),
        sectionsSpace: 2,
        centerSpaceRadius: 40,
      ),
    );
  }

  Widget _buildLegend() {
    final colors = [
      Colors.blue,
      Colors.red,
      Colors.green,
      Colors.orange,
      Colors.purple,
      Colors.pink,
      Colors.teal,
      Colors.amber,
    ];

    final entries = spending.entries.toList();

    return Wrap(
      spacing: 8,
      runSpacing: 8,
      alignment: WrapAlignment.center,
      children: entries.asMap().entries.map((entry) {
        final index = entry.key;

        final name = entry.value.key;

        final color = colors[index % colors.length];

        return Container(
          padding: EdgeInsets.symmetric(horizontal: 8, vertical: 4),
          decoration: BoxDecoration(
            color: color.withOpacity(0.1),
            borderRadius: BorderRadius.circular(12),
            border: Border.all(color: color.withOpacity(0.3)),
          ),
          child: Row(
            mainAxisSize: MainAxisSize.min,
            children: [
              Container(
                width: 12,
                height: 12,
                decoration: BoxDecoration(color: color, shape: BoxShape.circle),
              ),
              SizedBox(width: 4),
              Text(name, style: TextStyle(fontSize: 12)),
            ],
          ),
        );
      }).toList(),
    );
  }
}

10. Filter Dropdown Widget (lib/screens/widgets/filter_dropdown.dart)

lib/screens/widgets/filter_dropdown.dart 
import 'package:flutter/material.dart';

import 'package:flutter_riverpod/flutter_riverpod.dart';

import '../../models/expense.dart';

import '../../providers/expense_provider.dart';

class FilterDropdown extends ConsumerStatefulWidget {
  const FilterDropdown({super.key});

  @override
  ConsumerState<FilterDropdown> createState() => _FilterDropdownState();
}

class _FilterDropdownState extends ConsumerState<FilterDropdown> {
  String _selectedFilter = 'All';

  final List<String> _filterOptions = [
    'All',

    'Income',

    'Expense',
    ...Expense.categories,
  ];

  @override
  Widget build(BuildContext context) {
    return Container(
      decoration: BoxDecoration(
        color: Colors.white,
        borderRadius: BorderRadius.circular(8),
        border: Border.all(color: Colors.grey[300]!),
      ),
      padding: EdgeInsets.symmetric(horizontal: 8),
      child: DropdownButton<String>(
        value: _selectedFilter,
        isExpanded: true,
        underline: SizedBox(),
        icon: Icon(Icons.filter_list),
        items: _filterOptions.map((option) {
          return DropdownMenuItem(value: option, child: Text(option));
        }).toList(),
        onChanged: (value) {
          if (value == null) return;

          setState(() => _selectedFilter = value);

          final notifier = ref.read(expenseNotifierProvider.notifier);

          if (value == 'All') {
            notifier.clearFilters();
          } else if (value == 'Income') {
            notifier.setTypeFilter('income');
          } else if (value == 'Expense') {
            notifier.setTypeFilter('expense');
          } else {
            notifier.setCategoryFilter(value);
          }
        },
      ),
    );
  }
}
