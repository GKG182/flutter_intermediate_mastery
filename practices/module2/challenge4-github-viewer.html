GitHub Profile Viewer

Difficulty: Medium Module: API Integration Concepts: HTTP GET, User Search, Repository Listing Est. Time: 45-60 min 🐙 GitHub API

📋 Challenge Description

Build a GitHub Profile Viewer that allows users to search for GitHub profiles and view their repositories. Use the GitHub REST API to fetch user data and repository lists.

🌐 API Endpoints

GitHub - Get User

https://api.github.com/users/{username}

GitHub - List Repositories

https://api.github.com/users/{username}/repos

Returns user profile data including avatar, bio, followers, and repositories with stars, forks, and language.

💡 Note: GitHub's public API has rate limits (60 requests/hour for unauthenticated). You can increase this by adding a personal access token.

✅ Requirements

  • Search for a GitHub user by username
  • Display user profile (avatar, name, bio, followers, following)
  • List the user's repositories with stars, forks, and language
  • Show repository star and fork counts
  • Handle user not found errors gracefully
  • Add a loading indicator while fetching data
  • Use http package for API calls
  • Use FutureProvider for state management

📚 Resources

  • GitHub REST API
  • http package
  • Flutter Networking
Need a hint? Click here

💡 Hint: Create two models: GitHubUser and Repository . Use FutureProvider.autoDispose with a username parameter. Use StateProvider to track the username input. For the avatar, use NetworkImage or CachedNetworkImage .

class GitHubUser {
  final String login;

  final String? name;

  final String? avatarUrl;

  final String? bio;

  final int? publicRepos;

  final int? followers;

  final int? following;

  final String? location;

  final String? company;

  GitHubUser({
    required this.login,

    this.name,

    this.avatarUrl,

    this.bio,

    this.publicRepos,

    this.followers,

    this.following,

    this.location,

    this.company,
  });

  factory GitHubUser.fromJson(Map<String, dynamic> json) {
    return GitHubUser(
      login: json['login'] as String,
      name: json['name'] as String?,
      avatarUrl: json['avatar_url'] as String?,
      bio: json['bio'] as String?,
      publicRepos: json['public_repos'] as int?,
      followers: json['followers'] as int?,
      following: json['following'] as int?,
      location: json['location'] as String?,
      company: json['company'] as String?,
    );
  }
}

3. GitHub Provider (lib/providers/github_provider.dart)

lib/providers/github_provider.dart 
import 'package:flutter_riverpod/flutter_riverpod.dart';

import 'package:http/http.dart' as http;

import 'dart:convert';

import '../models/github_user.dart';

import '../models/repository.dart';

final usernameProvider = StateProvider<String>((ref) => 'flutter');

final userProvider = FutureProvider<GitHubUser?>.autoDispose((ref) async {
  final username = ref.watch(usernameProvider);

  if (username.isEmpty) return null;

  final url = Uri.parse('https://api.github.com/users/$username');

  final response = await http.get(url);

  if (response.statusCode == 200) {
    final data = json.decode(response.body);

    return GitHubUser.fromJson(data);
  } else if (response.statusCode == 404) {
    return null;
  } else {
    throw Exception('Failed to load user: ${response.statusCode}');
  }
});

final reposProvider = FutureProvider<List<Repository>>.autoDispose((ref) async {
  final username = ref.watch(usernameProvider);

  if (username.isEmpty) return [];

  final url = Uri.parse(
    'https://api.github.com/users/$username/repos?sort=updated&per_page=20',
  );

  final response = await http.get(url);

  if (response.statusCode == 200) {
    final List<dynamic> data = json.decode(response.body);

    return data.map((json) => Repository.fromJson(json)).toList();
  } else {
    throw Exception('Failed to load repositories: ${response.statusCode}');
  }
});

import 'package:flutter/material.dart';

import 'package:url_launcher/url_launcher.dart';

import '../models/repository.dart';

class RepositoryCard extends StatelessWidget {
  final Repository repo;

  const RepositoryCard({required this.repo, super.key});

  void _openRepo() async {
    if (repo.htmlUrl != null) {
      final url = Uri.parse(repo.htmlUrl!);

      if (await canLaunchUrl(url)) {
        await launchUrl(url);
      }
    }
  }

  @override
  Widget build(BuildContext context) {
    return Card(
      elevation: 1,
      margin: EdgeInsets.symmetric(vertical: 4),
      child: InkWell(
        onTap: _openRepo,
        borderRadius: BorderRadius.circular(8),
        child: Padding(
          padding: EdgeInsets.all(12),
          child: Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Row(
                children: [
                  Icon(Icons.repo, size: 16, color: Colors.grey[600]),
                  SizedBox(width: 8),
                  Expanded(
                    child: Text(
                      repo.name,
                      style: TextStyle(
                        fontSize: 16,
                        fontWeight: FontWeight.w600,
                        color: Colors.blue[700],
                      ),
                    ),
                  ),
                ],
              ),

              if (repo.description != null) ...[
                SizedBox(height: 4),
                Text(
                  repo.description!,
                  style: TextStyle(fontSize: 13, color: Colors.grey[700]),
                  maxLines: 2,

                ),

              SizedBox(height: 8),
              Row(

                      padding: EdgeInsets.symmetric(horizontal: 8, vertical: 2),

                        repo.language!,

                                  Colors.primaries.length],

                  _buildIconStat(Icons.fork_right, repo.forksCount.toString()),

6. Main Screen (lib/main.dart)

lib/main.dart 
import 'package:flutter/material.dart';

import 'package:flutter_riverpod/flutter_riverpod.dart';

import 'providers/github_provider.dart';

import 'widgets/user_profile.dart';

import 'widgets/repository_card.dart';

void main() {
  runApp(const ProviderScope(child: MyApp()));
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'GitHub Viewer',
      theme: ThemeData(
        colorScheme: ColorScheme.fromSeed(seedColor: Colors.purple),
        useMaterial3: true,
      ),
      home: const HomeScreen(),
    );
  }
}

class HomeScreen extends ConsumerStatefulWidget {
  const HomeScreen({super.key});

  @override
  ConsumerState<HomeScreen> createState() => _HomeScreenState();
}

class _HomeScreenState extends ConsumerState<HomeScreen> {
  final TextEditingController _controller = TextEditingController();

  void _searchUser() {
    final username = _controller.text.trim();

    if (username.isNotEmpty) {
      ref.read(usernameProvider.notifier).state = username;

      // Dismiss keyboard

      FocusScope.of(context).unfocus();
    }
  }

  @override
  Widget build(BuildContext context) {
    final userAsync = ref.watch(userProvider);

    final reposAsync = ref.watch(reposProvider);

    return Scaffold(
      appBar: AppBar(
        title: Text('🐙 GitHub Viewer'),
        backgroundColor: Colors.purple,
        foregroundColor: Colors.white,
      ),
      body: Padding(
        padding: EdgeInsets.all(16),
        child: Column(
          children: [
            // Search Bar
            Row(
              children: [
                Expanded(
                  child: TextField(
                    controller: _controller,
                    decoration: InputDecoration(
                      hintText: 'Enter GitHub username...',
                      border: OutlineInputBorder(
                        borderRadius: BorderRadius.circular(8),
                      ),
                      prefixIcon: Icon(Icons.search),
                      contentPadding: EdgeInsets.symmetric(horizontal: 16),
                    ),
                    onSubmitted: (_) => _searchUser(),
                  ),
                ),
                SizedBox(width: 8),
                ElevatedButton(onPressed: _searchUser, child: Text('Search')),
              ],
            ),
            SizedBox(height: 16),

            // Content
            Expanded(
              child: userAsync.when(
                loading: () => Center(
                  child: Column(
                    mainAxisAlignment: MainAxisAlignment.center,
                    children: [
                      CircularProgressIndicator(),
                      SizedBox(height: 16),
                      Text(
                        'Loading user...',
                        style: TextStyle(color: Colors.grey[600]),
                      ),
                    ],
                  ),
                ),
                error: (error, stack) => Center(
                  child: Column(
                    mainAxisAlignment: MainAxisAlignment.center,
                    children: [
                      Icon(
                        Icons.error_outline,
                        size: 64,
                        color: Colors.red[300],
                      ),
                      SizedBox(height: 16),
                      Text(
                        'Error loading user',
                        style: TextStyle(
                          fontSize: 18,
                          fontWeight: FontWeight.bold,
                        ),
                      ),
                      SizedBox(height: 8),
                      Text(
                        error.toString(),
                        textAlign: TextAlign.center,
                        style: TextStyle(color: Colors.grey[600]),
                      ),
                    ],
                  ),
                ),
                data: (user) {
                  if (user == null) {
                    return Center(
                      child: Column(
                        mainAxisAlignment: MainAxisAlignment.center,
                        children: [
                          Icon(
                            Icons.person_off,
                            size: 64,
                            color: Colors.grey[400],
                          ),
                          SizedBox(height: 16),
                          Text(
                            'User not found',
                            style: TextStyle(
                              fontSize: 18,
                              fontWeight: FontWeight.bold,
                            ),
                          ),
                          SizedBox(height: 8),
                          Text(
                            'Try a different username',
                            style: TextStyle(color: Colors.grey[500]),
                          ),
                        ],
                      ),
                    );
                  }

                  return SingleChildScrollView(
                    child: Column(
                      children: [
                        UserProfile(user: user),
                        SizedBox(height: 16),

                        // Repositories
                        Align(
                          alignment: Alignment.centerLeft,
                          child: Text(
                            '📁 Repositories',
                            style: TextStyle(
                              fontSize: 18,
                              fontWeight: FontWeight.bold,
                            ),
                          ),
                        ),
                        SizedBox(height: 8),
                        reposAsync.when(
                          loading: () =>
                              Center(child: CircularProgressIndicator()),
                          error: (error, stack) => Center(
                            child: Text(
                              'Error loading repos',
                              style: TextStyle(color: Colors.red[400]),
                            ),
                          ),
                          data: (repos) => repos.isEmpty
                              ? Padding(
                                  padding: EdgeInsets.all(16),
                                  child: Text(
                                    'No public repositories',
                                    style: TextStyle(color: Colors.grey[500]),
                                  ),
                                )
                              : ListView.builder(
                                  shrinkWrap: true,
                                  physics: NeverScrollableScrollPhysics(),
                                  itemCount: repos.length,
                                  itemBuilder: (context, index) {
                                    return RepositoryCard(repo: repos[index]);
                                  },
                                ),
                        ),
                        SizedBox(height: 16),
                      ],
                    ),
                  );
                },
              ),
            ),
          ],
        ),
      ),
    );
  }
}
