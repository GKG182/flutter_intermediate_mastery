Multi-Step Form State Manager

Difficulty: Hard Module: State Management Concepts: StateNotifier, Multi-Step Forms, Navigation Est. Time: 60-90 min

📋 Challenge Description

Build a Multi-Step Form State Manager — a Flutter app that guides users through a 3-step form flow (Personal Info → Address → Review & Submit) while preserving state across screens. Use StateNotifier to manage the form state and navigation to move between steps.

✅ Requirements

  • 3-step form flow:
    • Step 1: Personal Info (Name, Email, Phone)
    • Step 2: Address (Street, City, ZIP, Country)
    • Step 3: Review & Submit (show all data)
  • State persistence across all steps
  • Navigation between steps (Next/Back buttons)
  • Step indicator showing progress (e.g., Step 1/3)
  • Validation on each step before proceeding
  • Review screen showing all collected data
  • Submit with success message
  • Use StateNotifier for form state management
  • Clean, responsive UI with proper styling

📚 Resources

  • StateNotifierProvider
  • Flutter Navigation
  • flutter_riverpod
Need a hint? Click here

💡 Hint: Create a FormData class with all form fields. Use StateNotifier<FormData> to manage the state. For navigation, use Navigator.push to navigate to each step screen. Each screen should receive the FormData and update it using the notifier. Use a StateProvider<int> to track the current step.

2. Form Data Model (lib/models/form_data.dart)

lib/models/form_data.dart 
class FormData {
  // Step 1: Personal Info

  final String name;

  final String email;

  final String phone;

  // Step 2: Address

  final String street;

  final String city;

  final String zipCode;

  final String country;

  FormData({
    this.name = '',

    this.email = '',

    this.phone = '',

    this.street = '',

    this.city = '',

    this.zipCode = '',

    this.country = '',
  });

  FormData copyWith({
    String? name,
    String? email,
    String? phone,
    String? street,
    String? city,
    String? zipCode,
    String? country,
  }) {
    return FormData(
      name: name ?? this.name,
      email: email ?? this.email,
      phone: phone ?? this.phone,
      street: street ?? this.street,
      city: city ?? this.city,
      zipCode: zipCode ?? this.zipCode,
      country: country ?? this.country,
    );
  }

  bool get isStep1Valid =>
      name.isNotEmpty && email.contains('@') && phone.isNotEmpty;

  bool get isStep2Valid =>
      street.isNotEmpty &&
      city.isNotEmpty &&
      zipCode.isNotEmpty &&
      country.isNotEmpty;

  bool get isComplete => isStep1Valid && isStep2Valid;
}

3. Form Provider (lib/providers/form_provider.dart)

lib/providers/form_provider.dart 
import 'package:flutter_riverpod/flutter_riverpod.dart';

import '../models/form_data.dart';

class FormNotifier extends StateNotifier<FormData> {
  FormNotifier() : super(FormData());

  // Step 1: Personal Info

  void updateName(String value) {
    state = state.copyWith(name: value);
  }

  void updateEmail(String value) {
    state = state.copyWith(email: value);
  }

  void updatePhone(String value) {
    state = state.copyWith(phone: value);
  }

  // Step 2: Address

  void updateStreet(String value) {
    state = state.copyWith(street: value);
  }

  void updateCity(String value) {
    state = state.copyWith(city: value);
  }

  void updateZipCode(String value) {
    state = state.copyWith(zipCode: value);
  }

  void updateCountry(String value) {
    state = state.copyWith(country: value);
  }

  void resetForm() {
    state = FormData();
  }
}

final formProvider = StateNotifierProvider<FormNotifier, FormData>((ref) {
  return FormNotifier();
});

final currentStepProvider = StateProvider<int>((ref) => 0);

4. Step 1: Personal Info (lib/screens/personal_info_screen.dart)

lib/screens/personal_info_screen.dart 
import 'package:flutter/material.dart';

import 'package:flutter_riverpod/flutter_riverpod.dart';

import '../providers/form_provider.dart';

import 'address_screen.dart';

class PersonalInfoScreen extends ConsumerWidget {
  const PersonalInfoScreen({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final formState = ref.watch(formProvider);

    final notifier = ref.read(formProvider.notifier);

    return Scaffold(
      appBar: AppBar(
        title: Text('Step 1: Personal Info'),
        backgroundColor: Colors.purple,
        foregroundColor: Colors.white,
      ),
      body: Padding(
        padding: EdgeInsets.all(24),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            _buildStepIndicator(1, 3),
            SizedBox(height: 24),
            Text(
              'Personal Information',
              style: TextStyle(fontSize: 20, fontWeight: FontWeight.bold),
            ),
            SizedBox(height: 16),
            TextField(
              decoration: InputDecoration(
                labelText: 'Full Name',
                prefixIcon: Icon(Icons.person),
                border: OutlineInputBorder(),
              ),
              onChanged: (value) => notifier.updateName(value),
            ),
            SizedBox(height: 12),
            TextField(
              decoration: InputDecoration(
                labelText: 'Email Address',
                prefixIcon: Icon(Icons.email),
                border: OutlineInputBorder(),
                errorText:
                    formState.email.isNotEmpty && !formState.email.contains('@')
                    ? 'Please enter a valid email'
                    : null,
              ),
              onChanged: (value) => notifier.updateEmail(value),
            ),
            SizedBox(height: 12),
            TextField(
              decoration: InputDecoration(
                labelText: 'Phone Number',
                prefixIcon: Icon(Icons.phone),
                border: OutlineInputBorder(),
              ),
              onChanged: (value) => notifier.updatePhone(value),
            ),
            Spacer(),
            SizedBox(
              width: double.infinity,
              child: ElevatedButton(
                onPressed: formState.isStep1Valid
                    ? () {
                        ref.read(currentStepProvider.notifier).state = 1;
                        Navigator.push(
                          context,
                          MaterialPageRoute(
                            builder: (context) => AddressScreen(),
                          ),
                        );
                      }
                    : null,
                style: ElevatedButton.styleFrom(
                  padding: EdgeInsets.symmetric(vertical: 16),
                ),
                child: Text('Next →'),
              ),
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildStepIndicator(int current, int total) {
    return Row(
      children: List.generate(total, (index) {
        final isActive = index + 1 == current;

        return Expanded(
          child: Container(
            margin: EdgeInsets.symmetric(horizontal: 4),
            height: 4,
            decoration: BoxDecoration(
              color: isActive ? Colors.purple : Colors.grey[300],
              borderRadius: BorderRadius.circular(2),
            ),
          ),
        );
      }),
    );
  }
}
