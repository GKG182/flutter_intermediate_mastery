Theme Switcher Pro

Difficulty: Medium Module: State Management Concepts: StateProvider, ThemeData, Persistence Est. Time: 45-60 min

📋 Challenge Description

Build a Theme Switcher Pro — a Flutter app that allows users to toggle between Light and Dark themes with a smooth transition. The theme preference should persist even after the app is restarted, demonstrating state management combined with local persistence.

✅ Requirements

  • Light/Dark theme toggle with StateProvider
  • Theme preference persists after app restart (SharedPreferences)
  • Smooth theme transition animation
  • AppBar, background, text colors change with theme
  • Toggle button or switch with icon (sun/moon)
  • Show current theme name/status
  • Use StateProvider for theme state
  • Bonus: Add a third theme (e.g., System, Dark, Light)

📚 Resources

  • Riverpod Documentation
  • Flutter Themes

Need a hint? Click here

💡 Hint: Use StateProvider<ThemeMode> for the theme state. Store the theme preference in SharedPreferences and load it when the app starts. Use ProviderScope with overrides to initialize theme from stored value. For the transition, wrap your MaterialApp with AnimatedTheme .

2. Theme Provider (lib/providers/theme_provider.dart)

lib/providers/theme_provider.dart 
import 'package:flutter/material.dart';

import 'package:flutter_riverpod/flutter_riverpod.dart';

import 'package:shared_preferences/shared_preferences.dart';

final themeProvider = StateProvider<ThemeMode>((ref) {
  return ThemeMode.light;
});

final themePreferencesProvider = FutureProvider<ThemeMode?>((ref) async {
  final prefs = await SharedPreferences.getInstance();

  final themeString = prefs.getString('theme_mode');

  if (themeString == 'dark') return ThemeMode.dark;

  if (themeString == 'light') return ThemeMode.light;

  return null;
});

final themeNotifierProvider = Provider<ThemeNotifier>((ref) {
  return ThemeNotifier(ref);
});

class ThemeNotifier {
  final Ref ref;

  ThemeNotifier(this.ref);

  void toggleTheme() {
    final current = ref.read(themeProvider);

    final newTheme = current == ThemeMode.light
        ? ThemeMode.dark
        : ThemeMode.light;
    ref.read(themeProvider.notifier).state = newTheme;

    _saveTheme(newTheme);
  }

  void setTheme(ThemeMode mode) {
    ref.read(themeProvider.notifier).state = mode;

    _saveTheme(mode);
  }

  void _saveTheme(ThemeMode mode) async {
    final prefs = await SharedPreferences.getInstance();

    final value = mode == ThemeMode.dark ? 'dark' : 'light';

    await prefs.setString('theme_mode', value);
  }
}

// Theme data definitions

final lightTheme = ThemeData(
  useMaterial3: true,
  colorScheme: ColorScheme.fromSeed(
    seedColor: Colors.purple,
    brightness: Brightness.light,
  ),
);

final darkTheme = ThemeData(
  useMaterial3: true,
  colorScheme: ColorScheme.fromSeed(
    seedColor: Colors.purple,
    brightness: Brightness.dark,
  ),
);

3. Main App (lib/main.dart)

lib/main.dart 
import 'package:flutter/material.dart';

import 'package:flutter_riverpod/flutter_riverpod.dart';

import 'providers/theme_provider.dart';

import 'screens/home_screen.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends ConsumerStatefulWidget {
  const MyApp({super.key});

  @override
  ConsumerState<MyApp> createState() => _MyAppState();
}

class _MyAppState extends ConsumerState<MyApp> {
  bool _initialized = false;
  ThemeMode _initialTheme = ThemeMode.light;

  @override
  void initState() {
    super.initState();

    _loadTheme();
  }

  void _loadTheme() async {
    final prefs = await SharedPreferences.getInstance();

    final themeString = prefs.getString('theme_mode');

    if (themeString == 'dark') {
      _initialTheme = ThemeMode.dark;
    } else if (themeString == 'light') {
      _initialTheme = ThemeMode.light;
    }

    setState(() {
      _initialized = true;
    });

    // Also update the provider

    if (mounted) {
      ref.read(themeProvider.notifier).state = _initialTheme;
    }
  }

  @override
  Widget build(BuildContext context) {
    if (!_initialized) {
      return MaterialApp(
        home: Scaffold(body: Center(child: CircularProgressIndicator())),
      );
    }

    final themeMode = ref.watch(themeProvider);

    return AnimatedTheme(
      data: themeMode == ThemeMode.dark ? darkTheme : lightTheme,
      duration: Duration(milliseconds: 300),
      child: MaterialApp(
        title: 'Theme Switcher Pro',
        theme: lightTheme,
        darkTheme: darkTheme,
        themeMode: themeMode,
        home: const HomeScreen(),
      ),
    );
  }
}

4. Home Screen (lib/screens/home_screen.dart)

lib/screens/home_screen.dart 
import 'package:flutter/material.dart';

import 'package:flutter_riverpod/flutter_riverpod.dart';

import '../providers/theme_provider.dart';

class HomeScreen extends ConsumerWidget {
  const HomeScreen({super.key});

  @override
  Widget build(BuildContext context, WidgetRef ref) {
    final themeMode = ref.watch(themeProvider);

    final isDark = themeMode == ThemeMode.dark;

    return Scaffold(
      appBar: AppBar(
        title: Text('Theme Switcher Pro'),
        backgroundColor: Theme.of(context).colorScheme.primary,
        foregroundColor: Colors.white,
        actions: [
          IconButton(
            onPressed: () {
              ref.read(themeNotifierProvider).toggleTheme();
            },
            icon: Icon(isDark ? Icons.light_mode : Icons.dark_mode),
            tooltip: isDark ? 'Switch to Light' : 'Switch to Dark',
          ),
        ],
      ),
      body: Padding(
        padding: EdgeInsets.all(24),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            // Theme Status
            Container(
              padding: EdgeInsets.all(16),
              decoration: BoxDecoration(
                color: Theme.of(context).colorScheme.surfaceContainerHighest,
                borderRadius: BorderRadius.circular(12),
              ),
              child: Row(
                children: [
                  Icon(
                    isDark ? Icons.nightlight_round : Icons.wb_sunny,
                    size: 32,
                    color: isDark ? Colors.amber : Colors.orange,
                  ),
                  SizedBox(width: 16),
                  Column(
                    crossAxisAlignment: CrossAxisAlignment.start,
                    children: [
                      Text(
                        'Current Theme',
                        style: TextStyle(
                          fontSize: 14,
                          color: Theme.of(context).colorScheme.onSurfaceVariant,
                        ),
                      ),
                      Text(
                        isDark ? '🌙 Dark Mode' : '☀️ Light Mode',
                        style: TextStyle(
                          fontSize: 24,
                          fontWeight: FontWeight.bold,
                          color: Theme.of(context).colorScheme.onSurface,
                        ),
                      ),
                    ],
                  ),
                ],
              ),
            ),

            SizedBox(height: 24),

            // Toggle Switch
            Card(
              child: Padding(
                padding: EdgeInsets.all(16),
                child: Row(
                  mainAxisAlignment: MainAxisAlignment.spaceBetween,
                  children: [
                    Text(
                      'Dark Mode',
                      style: TextStyle(
                        fontSize: 16,
                        fontWeight: FontWeight.w500,
                      ),
                    ),
                    Switch(
                      value: isDark,
                      onChanged: (_) {
                        ref.read(themeNotifierProvider).toggleTheme();
                      },
                      activeColor: Colors.purple,
                    ),
                  ],
                ),
              ),
            ),

            SizedBox(height: 24),

            // Theme Preview Cards
            Text(
              'Theme Preview',
              style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
            ),
            SizedBox(height: 12),
            Row(
              children: [
                Expanded(
                  child: _buildPreviewCard(
                    context,
                    color: Colors.blue,
                    label: 'Primary',
                  ),
                ),
                SizedBox(width: 12),
                Expanded(
                  child: _buildPreviewCard(
                    context,
                    color: Colors.green,
                    label: 'Success',
                  ),
                ),
                SizedBox(width: 12),
                Expanded(
                  child: _buildPreviewCard(
                    context,
                    color: Colors.red,
                    label: 'Error',
                  ),
                ),
              ],
            ),

            SizedBox(height: 24),

            // Button Demo
            Text(
              'Button Styles',
              style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
            ),
            SizedBox(height: 12),
            Row(
              children: [
                ElevatedButton(onPressed: () {}, child: Text('Elevated')),
                SizedBox(width: 12),
                TextButton(onPressed: () {}, child: Text('Text')),
                SizedBox(width: 12),
                OutlinedButton(onPressed: () {}, child: Text('Outlined')),
              ],
            ),

            Spacer(),

            // Info text
            Center(
              child: Text(
                'Theme preference is saved automatically',
                style: TextStyle(
                  fontSize: 12,
                  color: Theme.of(context).colorScheme.onSurfaceVariant,
                ),
              ),
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildPreviewCard(
    BuildContext context, {

    required Color color,

    required String label,
  }) {
    return Container(
      padding: EdgeInsets.all(12),
      decoration: BoxDecoration(
        color: color.withOpacity(0.15),
        borderRadius: BorderRadius.circular(8),
        border: Border.all(color: color.withOpacity(0.3)),
      ),
      child: Column(
        children: [
          Icon(Icons.circle, color: color, size: 24),
          SizedBox(height: 4),
          Text(
            label,
            style: TextStyle(
              fontSize: 10,
              color: Theme.of(context).colorScheme.onSurface,
            ),
          ),
        ],
      ),
    );
  }
}
