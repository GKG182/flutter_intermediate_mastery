Smart Todo Manager

Difficulty: Easy Module: State Management Concepts: StateNotifierProvider, StateNotifier Est. Time: 30-45 min

📋 Challenge Description

Build a Smart Todo Manager — a fully functional todo app that demonstrates the power of Riverpod's StateNotifierProvider. Users should be able to add, delete, and toggle tasks, with filtering options to view All, Done, or Pending tasks.

✅ Requirements

  • Add new tasks with a text input and "Add" button
  • Each task has a checkbox to toggle completion status
  • Each task has a delete button to remove it
  • Filter tasks by: All, Done, Pending
  • Show the total count and pending count
  • Use StateNotifier and StateNotifierProvider from Riverpod
  • Tasks should have an ID and a timestamp
  • Clean, responsive UI with proper styling

📚 Resources

  • Riverpod Providers
  • StateNotifierProvider
  • Flutter Cookbook

💡 Hint: Define a Todo class with id, title, isDone, and createdAt. Use StateNotifier<List<Todo>> for the state. For filtering, create a Provider that watches the todo list and a filter variable to return the filtered list.

2. Todo Model (lib/models/todo.dart)

lib/models/todo.dart class Todo { final String id; final String title; final bool isDone; final DateTime createdAt; Todo({ required this.id, required this.title, this.isDone = false, DateTime? createdAt, }) : createdAt = createdAt ?? DateTime.now(); Todo copyWith({ String? id, String? title, bool? isDone, DateTime? createdAt, }) { return Todo( id: id ?? this.id, title: title ?? this.title, isDone: isDone ?? this.isDone, createdAt: createdAt ?? this.createdAt, ); } }

3. Todo Notifier (lib/providers/todo_provider.dart)

lib/providers/todo_provider.dart import 'package:flutter_riverpod/flutter_riverpod.dart'; import 'dart:math'; import '../models/todo.dart'; // Filter enum enum TodoFilter { all, done, pending } // StateNotifier for todos class TodoNotifier extends StateNotifier<List<Todo>> { TodoNotifier() : super([]); void addTodo(String title) { final todo = Todo( id: Random().nextInt(1000000).toString(), title: title, ); state = [...state, todo]; } void toggleTodo(String id) { state = state.map((todo) { if (todo.id == id) { return todo.copyWith(isDone: !todo.isDone); } return todo; }).toList(); } void deleteTodo(String id) { state = state.where((todo) => todo.id != id).toList(); } void clearCompleted() { state = state.where((todo) => !todo.isDone).toList(); } } // Providers final todoProvider = StateNotifierProvider<TodoNotifier, List<Todo>>((ref) { return TodoNotifier(); }); final filterProvider = StateProvider<TodoFilter>((ref) { return TodoFilter.all; }); final filteredTodosProvider = Provider<List<Todo>>((ref) { final todos = ref.watch(todoProvider); final filter = ref.watch(filterProvider); switch (filter) { case TodoFilter.done: return todos.where((todo) => todo.isDone).toList(); case TodoFilter.pending: return todos.where((todo) => !todo.isDone).toList(); case TodoFilter.all: default: return todos; } }); final todoStatsProvider = Provider<Map<String, int>>((ref) { final todos = ref.watch(todoProvider); final total = todos.length; final done = todos.where((t) => t.isDone).length; final pending = total - done; return {'total': total, 'done': done, 'pending': pending}; });

4. Todo Item Widget (lib/widgets/todo_item.dart)

lib/widgets/todo_item.dart import 'package:flutter/material.dart'; import 'package:flutter_riverpod/flutter_riverpod.dart'; import '../models/todo.dart'; import '../providers/todo_provider.dart'; class TodoItem extends ConsumerWidget { final Todo todo; const TodoItem({required this.todo, super.key}); @override Widget build(BuildContext context, WidgetRef ref) { return Card( margin: EdgeInsets.symmetric(vertical: 4), child: ListTile( leading: Checkbox( value: todo.isDone, onChanged: (_) { ref.read(todoProvider.notifier).toggleTodo(todo.id); }, ), title: Text( todo.title, style: TextStyle( decoration: todo.isDone ? TextDecoration.lineThrough : null, color: todo.isDone ? Colors.grey : Colors.black, ), ), trailing: Row( mainAxisSize: MainAxisSize.min, children: [ Text( _formatTime(todo.createdAt), style: TextStyle(fontSize: 10, color: Colors.grey[400]), ), IconButton( onPressed: () { ref.read(todoProvider.notifier).deleteTodo(todo.id); }, icon: Icon(Icons.delete_outline, color: Colors.red), ), ], ), ), ); } String _formatTime(DateTime time) { return '${time.hour.toString().padLeft(2, '0')}:${time.minute.toString().padLeft(2, '0')}'; } }

5. Main Screen (lib/main.dart)

lib/main.dart import 'package:flutter/material.dart'; import 'package:flutter_riverpod/flutter_riverpod.dart'; import 'providers/todo_provider.dart'; import 'widgets/todo_item.dart'; void main() { runApp(const ProviderScope(child: MyApp())); } class MyApp extends StatelessWidget { const MyApp({super.key}); @override Widget build(BuildContext context) { return MaterialApp( title: 'Smart Todo', theme: ThemeData( colorScheme: ColorScheme.fromSeed(seedColor: Colors.blue), useMaterial3: true, ), home: const HomeScreen(), ); } } class HomeScreen extends ConsumerStatefulWidget { const HomeScreen({super.key}); @override ConsumerState<HomeScreen> createState() => _HomeScreenState(); } class _HomeScreenState extends ConsumerState<HomeScreen> { final TextEditingController _controller = TextEditingController(); @override void dispose() { _controller.dispose(); super.dispose(); } void _addTodo() { final title = _controller.text.trim(); if (title.isNotEmpty) { ref.read(todoProvider.notifier).addTodo(title); _controller.clear(); } } @override Widget build(BuildContext context) { final filteredTodos = ref.watch(filteredTodosProvider); final stats = ref.watch(todoStatsProvider); final filter = ref.watch(filterProvider); return Scaffold( appBar: AppBar( title: Text('Smart Todo Manager'), backgroundColor: Colors.blue, foregroundColor: Colors.white, actions: [ if (stats['done']! > 0) IconButton( onPressed: () { ref.read(todoProvider.notifier).clearCompleted(); }, icon: Icon(Icons.cleaning_services), tooltip: 'Clear completed', ), ], ), body: Column( children: [ // Stats Container( padding: EdgeInsets.all(16), color: Colors.blue[50], child: Row( mainAxisAlignment: MainAxisAlignment.spaceEvenly, children: [ _buildStat('Total', stats['total']!), _buildStat('Done', stats['done']!, color: Colors.green), _buildStat('Pending', stats['pending']!, color: Colors.orange), ], ), ), // Filter chips Padding( padding: EdgeInsets.symmetric(horizontal: 16, vertical: 8), child: Row( children: [ _buildFilterChip('All', TodoFilter.all, filter), _buildFilterChip('Done', TodoFilter.done, filter), _buildFilterChip('Pending', TodoFilter.pending, filter), ], ), ), // Add todo input Padding( padding: EdgeInsets.all(16), child: Row( children: [ Expanded( child: TextField( controller: _controller, decoration: InputDecoration( hintText: 'Enter a task...', border: OutlineInputBorder(), contentPadding: EdgeInsets.symmetric(horizontal: 16), ), onSubmitted: (_) => _addTodo(), ), ), SizedBox(width: 8), ElevatedButton( onPressed: _addTodo, child: Text('Add'), ), ], ), ), // Todo list Expanded( child: filteredTodos.isEmpty ? Center( child: Column( mainAxisAlignment: MainAxisAlignment.center, children: [ Icon( filter == TodoFilter.done ? Icons.check_circle_outline : filter == TodoFilter.pending ? Icons.hourglass_empty : Icons.note_add, size: 64, color: Colors.grey[400], ), SizedBox(height: 16), Text( filter == TodoFilter.done ? 'No completed tasks' : filter == TodoFilter.pending ? 'All tasks completed! 🎉' : 'No tasks yet. Add one above!', style: TextStyle(fontSize: 16, color: Colors.grey[500]), ), ], ), ) : ListView.builder( padding: EdgeInsets.symmetric(horizontal: 16), itemCount: filteredTodos.length, itemBuilder: (context, index) { return TodoItem(todo: filteredTodos[index]); }, ), ), ], ), ); } Widget _buildStat(String label, int value, {Color? color}) { return Column( children: [ Text( value.toString(), style: TextStyle( fontSize: 24, fontWeight: FontWeight.bold, color: color ?? Colors.black, ), ), Text(label, style: TextStyle(fontSize: 12, color: Colors.grey[600])), ], ); } Widget _buildFilterChip(String label, TodoFilter value, TodoFilter current) { final isSelected = value == current; return Padding( padding: EdgeInsets.only(right: 8), child: ChoiceChip( label: Text(label), selected: isSelected, onSelected: (_) { ref.read(filterProvider.notifier).state = value; }, selectedColor: Colors.blue[200], ), ); } }

✅ Key Concepts Demonstrated

  • StateNotifier – Managing complex state with methods
  • StateNotifierProvider – Providing the notifier to the UI
  • Provider – Derived state (filtered todos, stats)
  • StateProvider – Simple state (filter selection)
  • Immutability – Creating new lists when updating state
