Dependency Injection Setup App

Difficulty: Medium Module: Clean Architecture Concepts: Dependency Injection, get_it, Modular Architecture, Environment Config Est. Time: 35-45 min 🏗️ DI

📋 Challenge Description

Build a Dependency Injection Setup App that demonstrates proper use of get_it for dependency injection with modular architecture and environment-based configurations.

✅ Requirements

  • Use get_it for dependency injection
  • Create separate modules for different app features
  • Implement environment-based configuration (dev, staging, prod)
  • Use factory and singleton registrations appropriately
  • Implement lazy loading for non-critical dependencies
  • Create abstract interfaces with multiple implementations
  • Use environment variables for API keys and URLs
  • Show dependency injection in action with a simple UI
  • Implement testing mocks using get_it
  • Handle service registration with proper order

📁 Module Structure

  • lib/
  • ├── core/
  • │ ├── di/ (injection.dart, modules/)
  • │ ├── config/ (environment.dart)
  • │ └── network/ (api_client.dart)
  • ├── features/
  • │ ├── auth/ (data/, domain/, presentation/)
  • │ └── dashboard/ (data/, domain/, presentation/)
  • └── main.dart

📚 Resources

  • get_it Package
  • flutter_dotenv
  • injectable (Code Generation)
  • mockito for testing
Need a hint? Click here

💡 Hint: Create separate modules for each feature (e.g., AuthModule, DashboardModule). Use registerFactory() for transient dependencies and registerSingleton() for shared services. Create an Environment class with dev , staging , and prod configs. Use registerLazySingleton() for services that shouldn't be created at startup.

3. API Client (lib/core/network/api_client.dart)

import 'package:dio/dio.dart';

import '../config/environment.dart';

abstract class ApiClient {
  Future<Response> get(String path);

  Future<Response> post(String path, {dynamic data});

  Future<Response> put(String path, {dynamic data});

  Future<Response> delete(String path);
}

class DioApiClient implements ApiClient {
  final Dio dio;

  final AppConfig config;

  DioApiClient(this.dio, this.config) {
    dio.options = BaseOptions(
      baseUrl: config.baseUrl,
      connectTimeout: Duration(seconds: 30),
      receiveTimeout: Duration(seconds: 30),
      headers: {
        'Authorization': 'Bearer ${config.apiKey}',

        'Content-Type': 'application/json',
      },
    );

    if (config.enableLogging) {
      dio.interceptors.add(
        LogInterceptor(requestBody: true, responseBody: true),
      );
    }
  }

  @override
  Future<Response> get(String path) {
    return dio.get(path);
  }

  @override
  Future<Response> post(String path, {dynamic data}) {
    return dio.post(path, data: data);
  }

  @override
  Future<Response> put(String path, {dynamic data}) {
    return dio.put(path, data: data);
  }

  @override
  Future<Response> delete(String path) {
    return dio.delete(path);
  }
}

import 'package:dartz/dartz.dart';

class AuthFailure {
  final String message;

  AuthFailure(this.message);
}

class User {
  final String id;

  final String email;

  final String name;

  User({required this.id, required this.email, required this.name});
}

abstract class AuthRepository {
  Future<Either<AuthFailure, User>> login(String email, String password);

  Future<Either<AuthFailure, void>> logout();

  Future<Either<AuthFailure, User>> getCurrentUser();

  Future<bool> isAuthenticated();
}

5. Auth Repository Implementation (lib/features/auth/data/repositories/auth_repository_impl.dart)

lib/features/auth/data/repositories/auth_repository_impl.dart 
import 'package:dartz/dartz.dart';

import 'package:shared_preferences/shared_preferences.dart';

import '../../../../core/network/api_client.dart';

import '../../domain/repositories/auth_repository.dart';

class AuthRepositoryImpl implements AuthRepository {
  final ApiClient apiClient;

  final SharedPreferences prefs;

  static const String _tokenKey = 'auth_token';

  AuthRepositoryImpl(this.apiClient, this.prefs);

  @override
  Future<Either<AuthFailure, User>> login(String email, String password) async {
    try {
      final response = await apiClient.post(
        '/auth/login',
        data: {'email': email, 'password': password},
      );

      if (response.statusCode == 200) {
        final data = response.data as Map<String, dynamic>;

        final user = User(
          id: data['id'],
          email: data['email'],
          name: data['name'],
        );

        await prefs.setString(_tokenKey, data['token']);

        return Right(user);
      }

      return Left(AuthFailure('Invalid credentials'));
    } catch (e) {
      return Left(AuthFailure(e.toString()));
    }
  }

  @override
  Future<Either<AuthFailure, void>> logout() async {
    try {
      await prefs.remove(_tokenKey);

      return const Right(null);
    } catch (e) {
      return Left(AuthFailure(e.toString()));
    }
  }

  @override
  Future<Either<AuthFailure, User>> getCurrentUser() async {
    try {
      final token = prefs.getString(_tokenKey);

      if (token == null) {
        return Left(AuthFailure('No user logged in'));
      }

      final response = await apiClient.get('/auth/me');

      final data = response.data as Map<String, dynamic>;

      return Right(
        User(id: data['id'], email: data['email'], name: data['name']),
      );
    } catch (e) {
      return Left(AuthFailure(e.toString()));
    }
  }

  @override
  Future<bool> isAuthenticated() async {
    return prefs.getString(_tokenKey) != null;
  }
}

6. DI Modules (lib/core/di/modules/)

lib/core/di/modules/network_module.dart 
import 'package:get_it/get_it.dart';

import 'package:dio/dio.dart';

import '../../config/environment.dart';

import '../../network/api_client.dart';

class NetworkModule {
  static void register(GetIt sl, AppConfig config) {
    // Register Dio as a factory (new instance each time)

    sl.registerFactory(() => Dio());

    // Register ApiClient based on environment

    if (config.enableMock) {
      sl.registerLazySingleton<ApiClient>(() => MockApiClient());
    } else {
      sl.registerLazySingleton<ApiClient>(() => DioApiClient(sl(), config));
    }
  }
}
lib/core/di/modules/storage_module.dart 
import 'package:get_it/get_it.dart';

import 'package:shared_preferences/shared_preferences.dart';

class StorageModule {
  static void register(GetIt sl) {
    // Register SharedPreferences as a singleton (one instance)

    sl.registerSingletonAsync(() async {
      return await SharedPreferences.getInstance();
    });
  }
}

final sl = GetIt.instance;

Future<void> initDependencyInjection(AppConfig config) async {
  // 1. Register config (singleton - always available)

  sl.registerSingleton<AppConfig>(config);

  // 2. Register storage (async - wait for completion)

  await StorageModule.register(sl);

  // 3. Register network (depends on config)

  NetworkModule.register(sl, config);

  // 4. Register features (depends on network and storage)

  AuthModule.register(sl);

9. Auth Provider (lib/features/auth/presentation/providers/auth_provider.dart)

lib/features/auth/presentation/providers/auth_provider.dart 
import 'package:flutter_riverpod/flutter_riverpod.dart';

import 'package:dartz/dartz.dart';

import '../../domain/repositories/auth_repository.dart';

import '../../../../core/di/injection.dart';

final authStateProvider = StreamProvider<User?>((ref) {
  return _getAuthStream();
});

Stream<User?> _getAuthStream() async* {
  final repository = sl<AuthRepository>();

  final result = await repository.getCurrentUser();

  yield result.fold((_) => null, (user) => user);
}

class AuthNotifier extends StateNotifier<AsyncValue<User?>> {
  final AuthRepository repository;

  AuthNotifier(this.repository) : super(const AsyncValue.loading()) {
    _loadCurrentUser();
  }

  Future<void> _loadCurrentUser() async {
    final result = await repository.getCurrentUser();
    result.fold(
      (_) => state = const AsyncValue.data(null),
      (user) => state = AsyncValue.data(user),
    );
  }

  Future<bool> login(String email, String password) async {
    state = const AsyncValue.loading();

    final result = await repository.login(email, password);

    return result.fold(
      (failure) {
        state = AsyncValue.error(failure.message, StackTrace.current);

        return false;
      },
      (user) {
        state = AsyncValue.data(user);

        return true;
      },
    );
  }

  Future<void> logout() async {
    state = const AsyncValue.loading();

    await repository.logout();
    state = const AsyncValue.data(null);
  }
}

final authNotifierProvider =
    StateNotifierProvider<AuthNotifier, AsyncValue<User?>>((ref) {
      return AuthNotifier(sl());
    });

class _LoginScreenState extends ConsumerState<LoginScreen> {
  final _emailController = TextEditingController();

  final _passwordController = TextEditingController();

  bool _isLoading = false;

  String? _errorMessage;

  void dispose() {
    _emailController.dispose();
    _passwordController.dispose();

    super.dispose();

  Future<void> _login() async {
    final email = _emailController.text.trim();

    final password = _passwordController.text.trim();

      setState(() => _errorMessage = 'Please fill in all fields');

      return;

      _isLoading = true;
      _errorMessage = null;

        .login(email, password);

      setState(() {
        _isLoading = false;
        _errorMessage = 'Invalid credentials';
      });
    }
