Clean Architecture Notes App

Difficulty: Hard Module: Clean Architecture Concepts: Clean Architecture, Use Cases, Repository Pattern, DI Est. Time: 60-75 min 🏗️ Clean Architecture

📋 Challenge Description

Build a notes app from scratch using Clean Architecture with separate data, domain, and presentation layers. This challenge focuses on proper separation of concerns, dependency injection, and use case implementation.

✅ Requirements

  • Create three layers : Data, Domain, and Presentation
  • Domain layer: Entities and Use Cases
  • Data layer: Repositories (Hive implementation)
  • Presentation layer: Riverpod with StateNotifier
  • Implement CRUD operations with use cases
  • Use dependency injection (get_it or constructor injection)
  • All dependencies should flow inward (data → domain → presentation)
  • Use failure handling with Either pattern or Result type
  • Demonstrate proper folder structure by layer
  • Write unit tests for use cases (optional but encouraged)

📁 Folder Structure

  • lib/
  • ├── data/
  • │ ├── datasources/ (Hive datasource)
  • │ ├── models/ (NoteModel)
  • │ └── repositories/ (NoteRepositoryImpl)
  • ├── domain/
  • │ ├── entities/ (Note entity)
  • │ ├── repositories/ (NoteRepository abstract)
  • │ └── usecases/ (CreateNote, GetNotes, etc.)
  • └── presentation/
  • ├── providers/ (NoteProvider)
  • ├── screens/ (HomeScreen, AddEditScreen)
  • └── widgets/ (NoteCard, etc.)

import 'package:equatable/equatable.dart';

import 'package:uuid/uuid.dart';

class Note extends Equatable {
  final String id;

  final String title;

  final String content;

  final DateTime createdAt;

  final DateTime updatedAt;

  Note({
    required this.id,

    required this.title,

    required this.content,

    required this.createdAt,

    required this.updatedAt,
  });

  Note.create({required String title, required String content})
    : this(
        id: Uuid().v4(),
        title: title,
        content: content,
        createdAt: DateTime.now(),
        updatedAt: DateTime.now(),
      );

  Note copyWith({String? title, String? content, DateTime? updatedAt}) {
    return Note(
      id: id,
      title: title ?? this.title,
      content: content ?? this.content,
      createdAt: createdAt,
      updatedAt: updatedAt ?? DateTime.now(),
    );
  }

  @override
  List<Object?> get props => [id, title, content, createdAt, updatedAt];

  String getFormattedDate() {
    return DateFormat.yMMMd().add_jm().format(updatedAt);
  }
}

class CreateNoteUseCase {

import 'package:hive_flutter/hive_flutter.dart';

import 'package:uuid/uuid.dart';

import '../../domain/entities/note.dart';

@HiveType(typeId: 0)
class NoteModel {
  @HiveField(0)
  final String id;

  @HiveField(1)
  final String title;

  @HiveField(2)
  final String content;

  @HiveField(3)
  final DateTime createdAt;

  @HiveField(4)
  final DateTime updatedAt;

  NoteModel({
    required this.id,

    required this.title,

    required this.content,

    required this.createdAt,

    required this.updatedAt,
  });

  factory NoteModel.fromDomain(Note note) {
    return NoteModel(
      id: note.id,
      title: note.title,
      content: note.content,
      createdAt: note.createdAt,
      updatedAt: note.updatedAt,
    );
  }

  Note toDomain() {
    return Note(
      id: id,
      title: title,
      content: content,
      createdAt: createdAt,
      updatedAt: updatedAt,
    );
  }

  factory NoteModel.create({required String title, required String content}) {
    final now = DateTime.now();

    return NoteModel(
      id: Uuid().v4(),
      title: title,
      content: content,
      createdAt: now,
      updatedAt: now,
    );
  }
}

7. Data Layer - Local Datasource

lib/data/datasources/note_local_datasource.dart 
import 'package:hive_flutter/hive_flutter.dart';

import '../models/note_model.dart';

class NoteLocalDatasource {
  final Box<NoteModel> box;

  NoteLocalDatasource(this.box);

  Future<List<NoteModel>> getAllNotes() async {
    return box.values.toList()
      ..sort((a, b) => b.updatedAt.compareTo(a.updatedAt));
  }

  Future<NoteModel?> getNoteById(String id) async {
    return box.values.firstWhere(
      (note) => note.id == id,
      orElse: () => throw Exception('Note not found'),
    );
  }

  Future<void> createNote(NoteModel note) async {
    await box.add(note);
  }

  Future<void> updateNote(NoteModel note) async {
    final index = box.values.toList().indexWhere((n) => n.id == note.id);

    if (index != -1) {
      await box.putAt(index, note);
    } else {
      throw Exception('Note not found');
    }
  }

  Future<void> deleteNote(String id) async {
    final index = box.values.toList().indexWhere((n) => n.id == id);

    if (index != -1) {
      await box.deleteAt(index);
    }
  }
}

8. Data Layer - Repository Implementation

lib/data/repositories/note_repository_impl.dart 
import 'package:dartz/dartz.dart';

import '../../domain/entities/note.dart';

import '../../domain/failures/failure.dart';

import '../../domain/repositories/note_repository.dart';

import '../datasources/note_local_datasource.dart';

import '../models/note_model.dart';

class NoteRepositoryImpl implements NoteRepository {
  final NoteLocalDatasource datasource;

  NoteRepositoryImpl(this.datasource);

  @override
  Future<Either<Failure, List<Note>>> getAllNotes() async {
    try {
      final notes = await datasource.getAllNotes();

      return Right(notes.map((n) => n.toDomain()).toList());
    } catch (e) {
      return Left(DatabaseFailure(e.toString()));
    }
  }

  @override
  Future<Either<Failure, Note>> getNoteById(String id) async {
    try {
      final note = await datasource.getNoteById(id);

      if (note == null) {
        return Left(NotFoundFailure('Note not found'));
      }

      return Right(note.toDomain());
    } catch (e) {
      return Left(DatabaseFailure(e.toString()));
    }
  }

  @override
  Future<Either<Failure, void>> createNote(Note note) async {
    try {
      final model = NoteModel.fromDomain(note);

      await datasource.createNote(model);

      return const Right(null);
    } catch (e) {
      return Left(DatabaseFailure(e.toString()));
    }
  }

  @override
  Future<Either<Failure, void>> updateNote(Note note) async {
    try {
      final model = NoteModel.fromDomain(
        note.copyWith(updatedAt: DateTime.now()),
      );

      await datasource.updateNote(model);

      return const Right(null);
    } catch (e) {
      return Left(DatabaseFailure(e.toString()));
    }
  }

  @override
  Future<Either<Failure, void>> deleteNote(String id) async {
    try {
      await datasource.deleteNote(id);

      return const Right(null);
    } catch (e) {
      return Left(DatabaseFailure(e.toString()));
    }
  }
}

final sl = GetIt.instance;

Future<void> initDependencies() async {
  // Register Hive box

  final box = await Hive.openBox<NoteModel>('notes');
  sl.registerLazySingleton(() => box);

  // Register datasource

  sl.registerLazySingleton(() => NoteLocalDatasource(sl()));

  // Register repository

  sl.registerLazySingleton<NoteRepository>(() => NoteRepositoryImpl(sl()));

  // Register use cases

  sl.registerLazySingleton(() => GetAllNotesUseCase(sl()));
  sl.registerLazySingleton(() => GetNoteByIdUseCase(sl()));
  sl.registerLazySingleton(() => CreateNoteUseCase(sl()));
  sl.registerLazySingleton(() => UpdateNoteUseCase(sl()));
  sl.registerLazySingleton(() => DeleteNoteUseCase(sl()));
}

lib/presentation/providers/note_provider.dart 
import 'package:flutter_riverpod/flutter_riverpod.dart';

import 'package:dartz/dartz.dart';

import '../../domain/entities/note.dart';

import '../../domain/failures/failure.dart';

import '../../domain/usecases/get_notes.dart';

import '../../di/injection.dart';

// Note state

class NoteState {
  final List<Note> notes;

  final bool isLoading;

  final String? error;

  NoteState({this.notes = const [], this.isLoading = false, this.error});

  NoteState copyWith({List<Note>? notes, bool? isLoading, String? error}) {
    return NoteState(
      notes: notes ?? this.notes,
      isLoading: isLoading ?? this.isLoading,
      error: error,
    );
  }
}

class NoteNotifier extends StateNotifier<NoteState> {
  final GetAllNotesUseCase _getAllNotes;

  final GetNoteByIdUseCase _getNoteById;

  final CreateNoteUseCase _createNote;

  final UpdateNoteUseCase _updateNote;

  final DeleteNoteUseCase _deleteNote;

  NoteNotifier(
    this._getAllNotes,

    this._getNoteById,

    this._createNote,

    this._updateNote,

    this._deleteNote,
  ) : super(NoteState());

  Future<void> loadNotes() async {
    state = state.copyWith(isLoading: true, error: null);

    final result = await _getAllNotes.call();
    result.fold(
      (failure) =>
          state = state.copyWith(isLoading: false, error: failure.message),
      (notes) =>
          state = state.copyWith(notes: notes, isLoading: false, error: null),
    );
  }

  Future<void> addNote(String title, String content) async {
    state = state.copyWith(isLoading: true, error: null);

    final note = Note.create(title: title, content: content);

    final result = await _createNote.call(note);
    result.fold(
      (failure) =>
          state = state.copyWith(isLoading: false, error: failure.message),
      (_) {
        state = state.copyWith(
          notes: [note, ...state.notes],
          isLoading: false,
          error: null,
        );
      },
    );
  }

  Future<void> editNote(Note note, String title, String content) async {
    state = state.copyWith(isLoading: true, error: null);

    final updatedNote = note.copyWith(title: title, content: content);

    final result = await _updateNote.call(updatedNote);
    result.fold(
      (failure) =>
          state = state.copyWith(isLoading: false, error: failure.message),
      (_) {
        final index = state.notes.indexWhere((n) => n.id == note.id);

        if (index != -1) {
          final updatedNotes = List<Note>.from(state.notes);
          updatedNotes[index] = updatedNote;
          state = state.copyWith(
            notes: updatedNotes,
            isLoading: false,
            error: null,
          );
        }
      },
    );
  }

  Future<void> deleteNote(String id) async {
    state = state.copyWith(isLoading: true, error: null);

    final result = await _deleteNote.call(id);
    result.fold(
      (failure) =>
          state = state.copyWith(isLoading: false, error: failure.message),
      (_) => state = state.copyWith(
        notes: state.notes.where((n) => n.id != id).toList(),
        isLoading: false,
        error: null,
      ),
    );
  }
}

final noteProvider = StateNotifierProvider<NoteNotifier, NoteState>((ref) {
  return NoteNotifier(sl(), sl(), sl(), sl(), sl());
});

  ConsumerState<HomeScreen> createState() => _HomeScreenState();

class _HomeScreenState extends ConsumerState<HomeScreen> {
  @override
  void initState() {
    super.initState();
    WidgetsBinding.instance.addPostFrameCallback((_) {
      ref.read(noteProvider.notifier).loadNotes();
    });
  }

  @override
  Widget build(BuildContext context) {
    final state = ref.watch(noteProvider);

    final notifier = ref.read(noteProvider.notifier);

    return Scaffold(
      appBar: AppBar(
        title: Text('📝 Clean Notes'),
        backgroundColor: Colors.indigo,
        foregroundColor: Colors.white,
        actions: [
          IconButton(
            onPressed: () => notifier.loadNotes(),
            icon: Icon(Icons.refresh),
            tooltip: 'Refresh',
          ),
        ],
      ),
      body: state.isLoading
          ? Center(child: CircularProgressIndicator())
          : state.error != null
          ? Center(
              child: Column(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  Icon(Icons.error_outline, size: 64, color: Colors.red[300]),
                  SizedBox(height: 16),
                  Text(
                    state.error!,
                    textAlign: TextAlign.center,
                    style: TextStyle(color: Colors.red),
                  ),
                  SizedBox(height: 16),
                  ElevatedButton(
                    onPressed: () => notifier.loadNotes(),
                    child: Text('Retry'),
                  ),
                ],
              ),
            )
          : state.notes.isEmpty
          ? _buildEmptyState()
          : ListView.builder(
              padding: EdgeInsets.all(16),
              itemCount: state.notes.length,
              itemBuilder: (context, index) {
                final note = state.notes[index];

                return Dismissible(
                  key: Key(note.id),
                  direction: DismissDirection.endToStart,
                  background: Container(
                    color: Colors.red,
                    alignment: Alignment.centerRight,
                    padding: EdgeInsets.only(right: 20),
                    child: Icon(Icons.delete, color: Colors.white),
                  ),
                  onDismissed: (_) {
                    notifier.deleteNote(note.id);
                    ScaffoldMessenger.of(
                      context,
                    ).showSnackBar(SnackBar(content: Text('Note deleted')));
                  },
                  child: NoteCard(
                    note: note,
                    onTap: () {
                      Navigator.push(
                        context,
                        MaterialPageRoute(
                          builder: (_) => AddEditNoteScreen(note: note),
                        ),
                      );
                    },
                  ),
                );
              },
            ),
      floatingActionButton: FloatingActionButton(
        onPressed: () {
          Navigator.push(
            context,
            MaterialPageRoute(builder: (_) => AddEditNoteScreen()),
          );
        },
        backgroundColor: Colors.indigo,
        foregroundColor: Colors.white,
        child: Icon(Icons.add),
      ),
    );
  }

  Widget _buildEmptyState() {
    return Center(
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          Icon(Icons.note_add, size: 80, color: Colors.grey[300]),
          SizedBox(height: 16),
          Text(

              fontSize: 20,

            style: TextStyle(color: Colors.grey[500]),
          ),
        ],
      ),
    );
  }
}
