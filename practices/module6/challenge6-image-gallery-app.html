Image Gallery App

Difficulty: Medium Module: Advanced Flutter Concepts: Unsplash API, Grid Layout, Search, Infinite Scroll Est. Time: 35-45 min ⚡ Advanced

📋 Challenge Description

Build an Image Gallery App that fetches and displays images from the Unsplash API with search, infinite scroll, and detail view.

✅ Requirements

  • Fetch images from Unsplash API with pagination
  • Display images in a responsive grid
  • Add search functionality for images
  • Implement infinite scrolling
  • Show image details on tap
  • Use cached_network_image for performance
  • Handle loading states and errors

📚 Resources

  • Unsplash API
  • cached_network_image
Need a hint? Click here

💡 Hint: Use cached_network_image for efficient image loading. For pagination, use ScrollController to detect when to load more. Register for a free Unsplash API key at unsplash.com/developers.

class ImageModel {
  final String id;

  final String url;

  final String thumb;

  final String? description;

  final String photographer;

  ImageModel({
    required this.id,

    required this.url,

    required this.thumb,

    this.description,

    required this.photographer,
  });

  factory ImageModel.fromJson(Map<String, dynamic> json) {
    return ImageModel(
      id: json['id'],
      url: json['urls']['regular'],
      thumb: json['urls']['thumb'],
      description: json['description'] as String?,
      photographer: json['user']['name'],
    );
  }
}

3. Gallery Screen

lib/presentation/screens/gallery_screen.dart 
import 'package:flutter/material.dart';

import 'package:flutter_riverpod/flutter_riverpod.dart';

import 'package:cached_network_image/cached_network_image.dart';

import '../../providers/image_provider.dart';

class GalleryScreen extends ConsumerStatefulWidget {
  const GalleryScreen({super.key});
  @override
  ConsumerState<GalleryScreen> createState() => _GalleryScreenState();
}

class _GalleryScreenState extends ConsumerState<GalleryScreen> {
  final ScrollController _scroll = ScrollController();

  final TextEditingController _search = TextEditingController();

  @override
  void initState() {
    super.initState();
    _scroll.addListener(() {
      if (_scroll.position.pixels >= _scroll.position.maxScrollExtent * 0.8) {
        ref.read(imageProvider.notifier).loadMore();
      }
    });
  }

  void dispose() {
    _scroll.dispose();
    _search.dispose();

    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final state = ref.watch(imageProvider);

    final notifier = ref.read(imageProvider.notifier);

    return Scaffold(
      appBar: AppBar(
        title: Text('🖼️ Image Gallery'),
        backgroundColor: Colors.pink,
        foregroundColor: Colors.white,
        bottom: PreferredSize(
          preferredSize: Size.fromHeight(70),
          child: Padding(
            padding: EdgeInsets.all(12),
            child: Row(
              children: [
                Expanded(
                  child: TextField(
                    controller: _search,
                    decoration: InputDecoration(
                      hintText: 'Search images...',
                      fillColor: Colors.white,
                      filled: true,
                      border: OutlineInputBorder(
                        borderRadius: BorderRadius.circular(8),
                        borderSide: BorderSide.none,
                      ),
                    ),
                    onSubmitted: (q) => notifier.search(q),
                  ),
                ),
                SizedBox(width: 8),
                ElevatedButton(
                  onPressed: () => notifier.search(_search.text),
                  style: ElevatedButton.styleFrom(
                    backgroundColor: Colors.white,
                    foregroundColor: Colors.pink,
                  ),
                  child: Text('Search'),
                ),
              ],
            ),
          ),
        ),
      ),
      body: RefreshIndicator(
        onRefresh: () async {
          notifier.refresh();
        },
        child: state.error != null
            ? Center(
                child: Column(
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: [
                    Icon(Icons.error, size: 64, color: Colors.red),
                    Text(state.error!),
                    ElevatedButton(
                      onPressed: notifier.refresh,
                      child: Text('Retry'),
                    ),
                  ],
                ),
              )
            : state.isLoading && state.images.isEmpty
            ? Center(child: CircularProgressIndicator())
            : GridView.builder(
                controller: _scroll,
                gridDelegate: SliverGridDelegateWithFixedCrossAxisCount(
                  crossAxisCount: 3,
                  crossAxisSpacing: 4,
                  mainAxisSpacing: 4,
                ),
                itemCount: state.images.length + (state.isLoading ? 1 : 0),
                itemBuilder: (context, index) {
                  if (index == state.images.length) {
                    return Center(child: CircularProgressIndicator());
                  }

                  final img = state.images[index];

                  return GestureDetector(
                    onTap: () => _showDetail(context, img),
                    child: Hero(
                      tag: img.id,
                      child: CachedNetworkImage(
                        imageUrl: img.thumb,
                        fit: BoxFit.cover,
                        placeholder: (_, __) =>
                            Container(color: Colors.grey[200]),
                        errorWidget: (_, __, ___) => Icon(Icons.error),
                      ),
                    ),
                  );
                },
              ),
      ),
    );
  }

  void _showDetail(BuildContext context, ImageModel img) {
    Navigator.push(
      context,
      MaterialPageRoute(
        builder: (_) => Scaffold(
          backgroundColor: Colors.black,
          appBar: AppBar(
            backgroundColor: Colors.transparent,
            foregroundColor: Colors.white,
            elevation: 0,
            title: Text(img.photographer, style: TextStyle(fontSize: 14)),
          ),
          body: Center(
            child: Hero(
              tag: img.id,
              child: InteractiveViewer(
                minScale: 0.5,
                maxScale: 4,
                child: CachedNetworkImage(imageUrl: img.url),
              ),
            ),
          ),
          bottomSheet: Container(
            padding: EdgeInsets.all(16),
            color: Colors.black.withOpacity(0.8),
            child: Text(
              img.description ?? 'No description',
              style: TextStyle(color: Colors.white),
            ),
          ),
        ),
      ),
    );
  }
}

4. Main

lib/main.dart 
import 'package:flutter/material.dart';

import 'package:flutter_riverpod/flutter_riverpod.dart';

import 'presentation/screens/gallery_screen.dart';

void main() => runApp(const ProviderScope(child: MyApp()));

class MyApp extends StatelessWidget {
  const MyApp({super.key});
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Image Gallery',
      theme: ThemeData(
        colorScheme: ColorScheme.fromSeed(seedColor: Colors.pink),
      ),
      home: const GalleryScreen(),
      debugShowCheckedModeBanner: false,
    );
  }
}
