Weather Animation Dashboard

Difficulty: Hard Module: Advanced Flutter Concepts: Custom Animations, Charting, API Integration, State Management Est. Time: 50-60 min ⚡ Advanced

📋 Challenge Description

Build a Weather Animation Dashboard with animated weather icons, temperature graphs, and smooth transitions. The app should fetch real weather data and display it with beautiful animations.

✅ Requirements

  • Fetch real-time weather data from OpenWeatherMap API
  • Display animated weather icons (sun, rain, clouds, etc.)
  • Show temperature graph with animated line chart
  • Display current weather with temperature, humidity, wind speed
  • Add search functionality for different cities
  • Show 5-day forecast with animated cards
  • Implement background color animations based on weather
  • Use animated transitions between weather states
  • Use Riverpod for state management
  • Handle loading states and errors gracefully

🌐 API Endpoint

OpenWeatherMap API

https://api.openweathermap.org/data/2.5/weather?q={city}&appid={API_KEY}
https://api.openweathermap.org/data/2.5/forecast?q={city}&appid={API_KEY}
💡 Note: You'll need to register for a free OpenWeatherMap API key at openweathermap.org

📚 Resources

  • http Package
  • Riverpod
  • fl_chart for Charts
  • Animated Text Kit
Need a hint? Click here

💡 Hint: Use AnimatedContainer for weather-based background color changes. Use AnimationController with Tween for icon animations. For the temperature graph, use fl_chart with smooth curves. Parse the weather condition from the API response to select appropriate animations. Use FutureProvider for async weather data fetching.

2. Weather Models (lib/models/weather.dart)

lib/models/weather.dart class Weather { final String cityName; final double temperature; final double feelsLike; final int humidity; final double windSpeed; final String description; final String iconCode; final DateTime timestamp; Weather({ required this.cityName, required this.temperature, required this.feelsLike, required this.humidity, required this.windSpeed, required this.description, required this.iconCode, required this.timestamp, }); factory Weather.fromJson(Map<String, dynamic> json) { return Weather( cityName: json['name'], temperature: (json['main']['temp'] as num).toDouble() - 273.15, feelsLike: (json['main']['feels_like'] as num).toDouble() - 273.15, humidity: json['main']['humidity'], windSpeed: (json['wind']['speed'] as num).toDouble(), description: json['weather'][0]['description'], iconCode: json['weather'][0]['icon'], timestamp: DateTime.fromMillisecondsSinceEpoch( json['dt'] * 1000, ), ); } String getIconUrl() { return 'https://openweathermap.org/img/wn/$iconCode@4x.png'; } bool get isDaytime { final hour = timestamp.hour; return hour >= 6 && hour < 18; } Color getBackgroundColor() { final weather = description.toLowerCase(); if (weather.contains('clear') || weather.contains('sun')) { return isDaytime ? Colors.blue[400]! : Colors.indigo[900]!; } else if (weather.contains('cloud')) { return isDaytime ? Colors.grey[500]! : Colors.grey[800]!; } else if (weather.contains('rain') || weather.contains('drizzle')) { return Colors.blueGrey[700]!; } else if (weather.contains('thunderstorm')) { return Colors.grey[900]!; } else if (weather.contains('snow')) { return Colors.blueGrey[200]!; } else { return Colors.blue[300]!; } } } class Forecast { final DateTime date; final double minTemp; final double maxTemp; final String description; final String iconCode; Forecast({ required this.date, required this.minTemp, required this.maxTemp, required this.description, required this.iconCode, }); factory Forecast.fromJson(Map<String, dynamic> json) { return Forecast( date: DateTime.fromMillisecondsSinceEpoch(json['dt'] * 1000), minTemp: (json['main']['temp_min'] as num).toDouble() - 273.15, maxTemp: (json['main']['temp_max'] as num).toDouble() - 273.15, description: json['weather'][0]['description'], iconCode: json['weather'][0]['icon'], ); } String getIconUrl() { return 'https://openweathermap.org/img/wn/$iconCode@2x.png'; } String getDayName() { return DateFormat.EEEE().format(date); } String getShortDayName() { return DateFormat.EEE().format(date); } }

3. Weather Provider (lib/providers/weather_provider.dart)

lib/providers/weather_provider.dart import 'package:flutter_riverpod/flutter_riverpod.dart'; import 'package:http/http.dart' as http; import 'dart:convert'; import '../models/weather.dart'; final cityProvider = StateProvider<String>((ref) => 'London'); class WeatherState { final Weather? weather; final List<Forecast> forecast; final bool isLoading; final String? error; WeatherState({ this.weather, this.forecast = const [], this.isLoading = false, this.error, }); WeatherState copyWith({ Weather? weather, List<Forecast>? forecast, bool? isLoading, String? error, }) { return WeatherState( weather: weather ?? this.weather, forecast: forecast ?? this.forecast, isLoading: isLoading ?? this.isLoading, error: error, ); } } class WeatherNotifier extends StateNotifier<WeatherState> { static const String apiKey = 'YOUR_OPENWEATHERMAP_API_KEY'; WeatherNotifier() : super(WeatherState()); Future<void> fetchWeather(String city) async { state = state.copyWith(isLoading: true, error: null); try { // Fetch current weather final weatherResponse = await http.get( Uri.parse('https://api.openweathermap.org/data/2.5/weather?q=$city&appid=$apiKey'), ); if (weatherResponse.statusCode != 200) { throw Exception('City not found'); } final weatherData = json.decode(weatherResponse.body); final weather = Weather.fromJson(weatherData); // Fetch forecast final forecastResponse = await http.get( Uri.parse('https://api.openweathermap.org/data/2.5/forecast?q=$city&appid=$apiKey'), ); final forecastData = json.decode(forecastResponse.body); final List<dynamic> list = forecastData['list']; // Get one forecast per day (at noon) Map<String, Forecast> dailyForecasts = {}; for (var item in list) { final forecast = Forecast.fromJson(item); final dateKey = DateFormat.yyyyMMdd().format(forecast.date); if (!dailyForecasts.containsKey(dateKey)) { dailyForecasts[dateKey] = forecast; } } final forecasts = dailyForecasts.values.toList(); final limitedForecast = forecasts.take(5).toList(); state = state.copyWith( weather: weather, forecast: limitedForecast, isLoading: false, error: null, ); } catch (e) { state = state.copyWith( isLoading: false, error: e.toString(), ); } } void refreshWeather() { final city = state.weather?.cityName ?? 'London'; fetchWeather(city); } } final weatherProvider = StateNotifierProvider<WeatherNotifier, WeatherState>((ref) { final notifier = WeatherNotifier(); final city = ref.watch(cityProvider); notifier.fetchWeather(city); return notifier; });

6. Animated Weather Icon (lib/presentation/widgets/weather_icon.dart)

lib/presentation/widgets/weather_icon.dart import 'package:flutter/material.dart'; import 'package:cached_network_image/cached_network_image.dart'; class AnimatedWeatherIcon extends StatefulWidget { final String iconCode; final double size; const AnimatedWeatherIcon({ required this.iconCode, required this.size, super.key, }); @override State<AnimatedWeatherIcon> createState() => _AnimatedWeatherIconState(); } class _AnimatedWeatherIconState extends State<AnimatedWeatherIcon> with SingleTickerProviderStateMixin { late AnimationController _controller; late Animation<double> _scaleAnimation; late Animation<double> _rotationAnimation; @override void initState() { super.initState(); _controller = AnimationController( vsync: this, duration: Duration(milliseconds: 800), )..forward(); _scaleAnimation = Tween<double>(begin: 0.5, end: 1.0) .animate(CurvedAnimation( parent: _controller, curve: Curves.elasticOut, )); _rotationAnimation = Tween<double>(begin: 0, end: 6.2832) .animate(CurvedAnimation( parent: _controller, curve: Curves.easeOut, )); } @override void didUpdateWidget(AnimatedWeatherIcon oldWidget) { super.didUpdateWidget(oldWidget); if (widget.iconCode != oldWidget.iconCode) { _controller.reset(); _controller.forward(); } } @override void dispose() { _controller.dispose(); super.dispose(); } @override Widget build(BuildContext context) { return AnimatedBuilder( animation: _controller, builder: (context, child) { return Transform.scale( scale: _scaleAnimation.value, child: Transform.rotate( angle: _rotationAnimation.value * 0.5, child: child, ), ); }, child: Image.network( 'https://openweathermap.org/img/wn/${widget.iconCode}@4x.png', width: widget.size, height: widget.size, fit: BoxFit.contain, errorBuilder: (context, error, stackTrace) { return Icon( Icons.wb_sunny, size: widget.size, color: Colors.yellow[300], ); }, ), ); } }
