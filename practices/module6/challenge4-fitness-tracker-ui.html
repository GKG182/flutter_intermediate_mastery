Fitness Tracker UI

Difficulty: Hard Module: Advanced Flutter Concepts: Animated Charts, Progress Rings, Step Counter, Gestures Est. Time: 50-60 min ⚡ Advanced

📋 Challenge Description

Build a Fitness Tracker UI with animated charts, step counter, workout progress rings, and gesture-based interactions. The app should provide a visually engaging fitness dashboard experience.

✅ Requirements

  • Display step counter with animated number
  • Show progress rings for daily goals (steps, calories, distance)
  • Create animated bar chart for weekly activity
  • Add workout cards with swipe gestures
  • Show calorie tracking with animated progress
  • Implement gesture-based navigation between views
  • Add pull-to-refresh functionality
  • Use custom animations for transitions
  • Support dark/light theme
  • Use Riverpod for state management

📦 Sample Data

  • Steps: 8,432 (goal: 10,000)
  • Calories: 2,150 (goal: 2,500)
  • Distance: 6.2 km (goal: 8.0 km)
  • Weekly Activity: Mon-Sun step counts
  • Workouts: Running, Cycling, Swimming, Yoga

📚 Resources

  • Riverpod
  • fl_chart for Charts
  • Flutter Animations Guide
  • percent_indicator
Need a hint? Click here

💡 Hint: Use AnimatedContainer for smooth transitions. For progress rings, use CustomPaint with AnimationController. For the bar chart, use fl_chart with animated bars. Use Dismissible for swipe-to-remove on workout cards. Use AnimationController with Tween for number animations.

lib/providers/fitness_provider.dart import 'package:flutter_riverpod/flutter_riverpod.dart'; import '../models/fitness.dart'; class FitnessState { final FitnessStats stats; final List<DailyActivity> weeklyActivity; final List<Workout> workouts; FitnessState({ required this.stats, required this.weeklyActivity, required this.workouts, }); FitnessState copyWith({ FitnessStats? stats, List<DailyActivity>? weeklyActivity, List<Workout>? workouts, }) { return FitnessState( stats: stats ?? this.stats, weeklyActivity: weeklyActivity ?? this.weeklyActivity, workouts: workouts ?? this.workouts, ); } factory FitnessState.initial() { return FitnessState( stats: FitnessStats( steps: 8432, goalSteps: 10000, calories: 2150, goalCalories: 2500, distance: 6.2, goalDistance: 8.0, ), weeklyActivity: FitnessStats.sampleWeeklyActivity, workouts: FitnessStats.sampleWorkouts, ); } } class FitnessNotifier extends StateNotifier<FitnessState> { FitnessNotifier() : super(FitnessState.initial()); void updateStats({ int? steps, int? calories, double? distance, }) { final currentStats = state.stats; state = state.copyWith( stats: FitnessStats( steps: steps ?? currentStats.steps, goalSteps: currentStats.goalSteps, calories: calories ?? currentStats.calories, goalCalories: currentStats.goalCalories, distance: distance ?? currentStats.distance, goalDistance: currentStats.goalDistance, ), ); } void addWorkout(Workout workout) { state = state.copyWith( workouts: [...state.workouts, workout], ); } void removeWorkout(String id) { state = state.copyWith( workouts: state.workouts.where((w) => w.id != id).toList(), ); } void refreshStats() { // Simulate refreshing with random new values final random = DateTime.now().millisecond % 1000; state = state.copyWith( stats: FitnessStats( steps: 7000 + random % 4000, goalSteps: 10000, calories: 1800 + random % 800, goalCalories: 2500, distance: 5.0 + (random % 400) / 100, goalDistance: 8.0, ), ); } } final fitnessProvider = StateNotifierProvider<FitnessNotifier, FitnessState>((ref) { return FitnessNotifier(); });

lib/presentation/widgets/progress_ring.dart import 'package:flutter/material.dart'; class ProgressRing extends StatelessWidget { final double value; final String label; final Color color; final double size; final AnimationController animation; const ProgressRing({ required this.value, required this.label, required this.color, required this.size, required this.animation, super.key, }); @override Widget build(BuildContext context) { return AnimatedBuilder( animation: animation, builder: (context, child) { final animatedValue = animation.value * value; return SizedBox( width: size, height: size, child: Stack( children: [ CustomPaint( painter: _ProgressRingPainter( value: animatedValue.clamp(0.0, 1.0), color: color, backgroundColor: Colors.grey[200]!, ), size: Size(size, size), ), Center( child: Column( mainAxisAlignment: MainAxisAlignment.center, children: [ Text( '${(animatedValue * 100).clamp(0, 100).toInt()}%', style: TextStyle( fontSize: 14, fontWeight: FontWeight.bold, color: color, ), ), Text( label, style: TextStyle( fontSize: 10, color: Colors.grey[600], ), ), ], ), ), ], ), ); }, ); } } class _ProgressRingPainter extends CustomPainter { final double value; final Color color; final Color backgroundColor; _ProgressRingPainter({ required this.value, required this.color, required this.backgroundColor, }); @override void paint(Canvas canvas, Size size) { final center = Offset(size.width / 2, size.height / 2); final radius = size.width / 2 - 4; final strokeWidth = 6.0; // Background ring final backgroundPaint = Paint() ..color = backgroundColor ..strokeWidth = strokeWidth ..style = PaintingStyle.stroke; canvas.drawArc( Rect.fromCircle(center: center, radius: radius), -pi / 2, 2 * pi, false, backgroundPaint, ); // Progress ring final progressPaint = Paint() ..color = color ..strokeWidth = strokeWidth ..strokeCap = StrokeCap.round ..style = PaintingStyle.stroke; canvas.drawArc( Rect.fromCircle(center: center, radius: radius), -pi / 2, 2 * pi * value, false, progressPaint, ); } @override bool shouldRepaint(_ProgressRingPainter oldDelegate) { return oldDelegate.value != value;
