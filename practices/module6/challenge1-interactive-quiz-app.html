Interactive Quiz App

Difficulty: Medium Module: Advanced Flutter Concepts: Animations, State Management, Timer, Score Tracking Est. Time: 40-50 min ⚡ Advanced

📋 Challenge Description

Build an Interactive Quiz App with animated transitions, score tracking, progress indicators, and timed questions. The app should provide a fun and engaging quiz experience with smooth animations.

✅ Requirements

  • Display multiple-choice questions with 4 options
  • Show animated progress indicator for quiz progress
  • Add question transition animations (slide/fade)
  • Implement score tracking with correct/incorrect answers
  • Show feedback animations for correct/incorrect answers
  • Add a timer for each question (15-30 seconds)
  • Display results screen with final score and statistics
  • Include restart functionality with animation
  • Use Riverpod for state management
  • Support multiple categories or question sets

📦 Sample Quiz Data

  • Category: Flutter, Dart, Programming, General Knowledge
  • Questions: 10 questions per quiz
  • Time per question: 20 seconds
  • Scoring: +10 for correct, 0 for incorrect

📚 Resources

  • Flutter Animations Guide
  • Riverpod
  • Flutter Tabs & Pages
  • Hero Animations
Need a hint? Click here

💡 Hint: Use AnimatedContainer or AnimatedSwitcher for question transitions. For the timer, use Timer.periodic() with a state variable. For feedback animations, use AnimatedScale or AnimatedOpacity with a delay. Store quiz data as a list of Question objects with id, text, options, and correctAnswerIndex.

3. Quiz Provider (lib/providers/quiz_provider.dart)

lib/providers/quiz_provider.dart import 'dart:async'; import 'package:flutter_riverpod/flutter_riverpod.dart'; import '../models/question.dart'; class QuizState { final List<Question> questions; final int currentQuestionIndex; final int score; final List<bool> answers; final bool isAnswered; final bool isQuizComplete; final int selectedOption; final int timeRemaining; final bool showFeedback; QuizState({ required this.questions, required this.currentQuestionIndex, required this.score, required this.answers, required this.isAnswered, required this.isQuizComplete, required this.selectedOption, required this.timeRemaining, required this.showFeedback, }); factory QuizState.initial({List<Question>? questions}) { return QuizState( questions: questions ?? Question.sampleQuestions, currentQuestionIndex: 0, score: 0, answers: [], isAnswered: false, isQuizComplete: false, selectedOption: -1, timeRemaining: 20, showFeedback: false, ); } QuizState copyWith({ List<Question>? questions, int? currentQuestionIndex, int? score, List<bool>? answers, bool? isAnswered, bool? isQuizComplete, int? selectedOption, int? timeRemaining, bool? showFeedback, }) { return QuizState( questions: questions ?? this.questions, currentQuestionIndex: currentQuestionIndex ?? this.currentQuestionIndex, score: score ?? this.score, answers: answers ?? this.answers, isAnswered: isAnswered ?? this.isAnswered, isQuizComplete: isQuizComplete ?? this.isQuizComplete, selectedOption: selectedOption ?? this.selectedOption, timeRemaining: timeRemaining ?? this.timeRemaining, showFeedback: showFeedback ?? this.showFeedback, ); } int get totalQuestions => questions.length; int get correctAnswers => answers.where((a) => a).length; double get progress => totalQuestions > 0 ? (currentQuestionIndex + (isAnswered ? 1 : 0)) / totalQuestions : 0.0; } class QuizNotifier extends StateNotifier<QuizState> { Timer? _timer; QuizNotifier() : super(QuizState.initial()) { _startTimer(); } void _startTimer() { _timer?.cancel(); state = state.copyWith(timeRemaining: 20); _timer = Timer.periodic(Duration(seconds: 1), (timer) { if (state.isAnswered || state.isQuizComplete) { return; } if (state.timeRemaining > 0) { state = state.copyWith(timeRemaining: state.timeRemaining - 1); } else { // Time's up - mark as incorrect _handleTimeUp(); } }); } void _handleTimeUp() { if (state.isAnswered) return; // Mark as answered with no selection state = state.copyWith( isAnswered: true, showFeedback: true, selectedOption: -1, answers: [...state.answers, false], ); // Auto advance after delay Future.delayed(Duration(seconds: 1), () { if (!state.isQuizComplete && state.currentQuestionIndex < state.totalQuestions - 1) { _nextQuestion(); } }); } void selectAnswer(int selectedIndex) { if (state.isAnswered || state.isQuizComplete) return; final question = state.questions[state.currentQuestionIndex]; final isCorrect = selectedIndex == question.correctAnswerIndex; // Show feedback immediately state = state.copyWith( isAnswered: true, showFeedback: true, selectedOption: selectedIndex, score: isCorrect ? state.score + 10 : state.score, answers: [...state.answers, isCorrect], timeRemaining: 0, ); _timer?.cancel(); // Auto advance after delay Future.delayed(Duration(milliseconds: 800), () { if (!state.isQuizComplete && state.currentQuestionIndex < state.totalQuestions - 1) { _nextQuestion(); } }); } void _nextQuestion() { if (state.currentQuestionIndex < state.totalQuestions - 1) { state = state.copyWith( currentQuestionIndex: state.currentQuestionIndex + 1, isAnswered: false, selectedOption: -1, timeRemaining: 20, showFeedback: false, ); _startTimer(); } else { state = state.copyWith(isQuizComplete: true); } } void restartQuiz() { _timer?.cancel(); state = QuizState.initial(questions: state.questions); _startTimer(); } @override void dispose() { _timer?.cancel(); super.dispose(); } } final quizProvider = StateNotifierProvider<QuizNotifier, QuizState>((ref) { return QuizNotifier(); });
