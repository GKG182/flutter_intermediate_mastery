Interactive Quiz App

Difficulty: Medium Module: Advanced Flutter Concepts: Animations, State Management, Timer, Score Tracking Est. Time: 40-50 min ⚡ Advanced

📋 Challenge Description

Build an Interactive Quiz App with animated transitions, score tracking, progress indicators, and timed questions. The app should provide a fun and engaging quiz experience with smooth animations.

✅ Requirements

  • Display multiple-choice questions with 4 options
  • Show animated progress indicator for quiz progress
  • Add question transition animations (slide/fade)
  • Implement score tracking with correct/incorrect answers
  • Show feedback animations for correct/incorrect answers
  • Add a timer for each question (15-30 seconds)
  • Display results screen with final score and statistics
  • Include restart functionality with animation
  • Use Riverpod for state management
  • Support multiple categories or question sets

📦 Sample Quiz Data

  • Category: Flutter, Dart, Programming, General Knowledge
  • Questions: 10 questions per quiz
  • Time per question: 20 seconds
  • Scoring: +10 for correct, 0 for incorrect

📚 Resources

  • Flutter Animations Guide
  • Riverpod
  • Flutter Tabs & Pages
  • Hero Animations
Need a hint? Click here

💡 Hint: Use AnimatedContainer or AnimatedSwitcher for question transitions. For the timer, use Timer.periodic() with a state variable. For feedback animations, use AnimatedScale or AnimatedOpacity with a delay. Store quiz data as a list of Question objects with id , text , options , and correctAnswerIndex .

import 'dart:async';

import 'package:flutter_riverpod/flutter_riverpod.dart';

import '../models/question.dart';

class QuizState {
  final List<Question> questions;

  final int currentQuestionIndex;

  final int score;

  final List<bool> answers;

  final bool isAnswered;

  final bool isQuizComplete;

  final int selectedOption;

  final int timeRemaining;

  final bool showFeedback;

  QuizState({
    required this.questions,

    required this.currentQuestionIndex,

    required this.score,

    required this.answers,

    required this.isAnswered,

    required this.isQuizComplete,

    required this.selectedOption,

    required this.timeRemaining,

    required this.showFeedback,
  });

  factory QuizState.initial({List<Question>? questions}) {
    return QuizState(
      questions: questions ?? Question.sampleQuestions,
      currentQuestionIndex: 0,
      score: 0,
      answers: [],
      isAnswered: false,
      isQuizComplete: false,
      selectedOption: -1,
      timeRemaining: 20,
      showFeedback: false,
    );
  }

  QuizState copyWith({
    List<Question>? questions,

    int? currentQuestionIndex,

    int? score,
    List<bool>? answers,

    bool? isAnswered,

    bool? isQuizComplete,

    int? selectedOption,

    int? timeRemaining,

    bool? showFeedback,
  }) {
    return QuizState(
      questions: questions ?? this.questions,
      currentQuestionIndex: currentQuestionIndex ?? this.currentQuestionIndex,
      score: score ?? this.score,
      answers: answers ?? this.answers,
      isAnswered: isAnswered ?? this.isAnswered,
      isQuizComplete: isQuizComplete ?? this.isQuizComplete,
      selectedOption: selectedOption ?? this.selectedOption,
      timeRemaining: timeRemaining ?? this.timeRemaining,
      showFeedback: showFeedback ?? this.showFeedback,
    );
  }

  int get totalQuestions => questions.length;

  int get correctAnswers => answers.where((a) => a).length;

  double get progress => totalQuestions > 0
      ? (currentQuestionIndex + (isAnswered ? 1 : 0)) / totalQuestions
      : 0.0;
}

class QuizNotifier extends StateNotifier<QuizState> {
  Timer? _timer;

  QuizNotifier() : super(QuizState.initial()) {
    _startTimer();
  }

  void _startTimer() {
    _timer?.cancel();
    state = state.copyWith(timeRemaining: 20);
    _timer = Timer.periodic(Duration(seconds: 1), (timer) {
      if (state.isAnswered || state.isQuizComplete) {
        return;
      }

      if (state.timeRemaining > 0) {
        state = state.copyWith(timeRemaining: state.timeRemaining - 1);
      } else {
        // Time's up - mark as incorrect

        _handleTimeUp();
      }
    });
  }

  void _handleTimeUp() {
    if (state.isAnswered) return;

    // Mark as answered with no selection

    state = state.copyWith(
      isAnswered: true,
      showFeedback: true,
      selectedOption: -1,
      answers: [...state.answers, false],
    );

    // Auto advance after delay

    Future.delayed(Duration(seconds: 1), () {
      if (!state.isQuizComplete &&
          state.currentQuestionIndex < state.totalQuestions - 1) {
        _nextQuestion();
      }
    });
  }

  void selectAnswer(int selectedIndex) {
    if (state.isAnswered || state.isQuizComplete) return;

    final question = state.questions[state.currentQuestionIndex];

    final isCorrect = selectedIndex == question.correctAnswerIndex;

    // Show feedback immediately

    state = state.copyWith(
      isAnswered: true,
      showFeedback: true,
      selectedOption: selectedIndex,
      score: isCorrect ? state.score + 10 : state.score,
      answers: [...state.answers, isCorrect],
      timeRemaining: 0,
    );

    _timer?.cancel();

    // Auto advance after delay

    Future.delayed(Duration(milliseconds: 800), () {
      if (!state.isQuizComplete &&
          state.currentQuestionIndex < state.totalQuestions - 1) {
        _nextQuestion();
      }
    });
  }

  void _nextQuestion() {
    if (state.currentQuestionIndex < state.totalQuestions - 1) {
      state = state.copyWith(
        currentQuestionIndex: state.currentQuestionIndex + 1,
        isAnswered: false,
        selectedOption: -1,
        timeRemaining: 20,
        showFeedback: false,
      );

      _startTimer();
    } else {
      state = state.copyWith(isQuizComplete: true);
    }
  }

  void restartQuiz() {
    _timer?.cancel();
    state = QuizState.initial(questions: state.questions);

    _startTimer();
  }

  @override
  void dispose() {
    _timer?.cancel();

    super.dispose();
  }
}

final quizProvider = StateNotifierProvider<QuizNotifier, QuizState>((ref) {
  return QuizNotifier();
});

              style: TextStyle(

                fontWeight: FontWeight.w600,
                height: 1.3,
              ),
            ),
            SizedBox(height: 20),

            // Options
            Expanded(
              child: ListView.builder(
                itemCount: widget.question.options.length,
                itemBuilder: (context, index) {
                  final option = widget.question.options[index];

                  final isSelected = widget.selectedOption == index;

                  final isCorrect = index == widget.question.correctAnswerIndex;

                  final showCorrect = widget.isAnswered && isCorrect;

                  final showWrong =
                      widget.isAnswered && isSelected && !isCorrect;

                  Color backgroundColor;
                  Color borderColor;

                  if (showCorrect) {
                    backgroundColor = Colors.green.withOpacity(0.15);
                    borderColor = Colors.green;
                  } else if (showWrong) {
                    backgroundColor = Colors.red.withOpacity(0.15);
                    borderColor = Colors.red;
                  } else if (isSelected && !widget.isAnswered) {
                    backgroundColor = Colors.deepPurple.withOpacity(0.1);
                    borderColor = Colors.deepPurple;
                  } else {
                    backgroundColor = Colors.grey[50]!;
                    borderColor = Colors.grey[300]!;
                  }

                  return AnimatedContainer(
                    duration: Duration(milliseconds: 300),
                    margin: EdgeInsets.only(bottom: 12),
                    decoration: BoxDecoration(
                      color: backgroundColor,
                      borderRadius: BorderRadius.circular(12),

                      leading: CircleAvatar(

class AnimatedProgressBar extends StatefulWidget {
  final double progress;

  const AnimatedProgressBar({required this.progress, super.key});

  @override
  State<AnimatedProgressBar> createState() => _AnimatedProgressBarState();
}

class _AnimatedProgressBarState extends State<AnimatedProgressBar>
    with SingleTickerProviderStateMixin {
  late AnimationController _controller;
  late Animation<double> _animation;

  @override
  void initState() {
    super.initState();
    _controller = AnimationController(
      vsync: this,
      duration: Duration(milliseconds: 500),
    );
    _animation = Tween<double>(
      begin: 0,
      end: widget.progress,
    ).animate(_controller)..addListener(() => setState(() {}));
    _controller.forward();
  }

  @override
  void didUpdateWidget(AnimatedProgressBar oldWidget) {
    super.didUpdateWidget(oldWidget);

    if (widget.progress != oldWidget.progress) {
      _animation = Tween<double>(
        begin: _animation.value,
        end: widget.progress,
      ).animate(_controller)..addListener(() => setState(() {}));
      _controller.reset();
      _controller.forward();
    }
  }

  @override
  void dispose() {
    _controller.dispose();

    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Padding(
      padding: EdgeInsets.symmetric(horizontal: 16),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          Row(
            children: [
              Text(
                'Progress',
                style: TextStyle(
                  fontSize: 12,

class TimerWidget extends StatefulWidget {
  final int timeRemaining;

  final bool isAnswered;

  const TimerWidget({
    required this.timeRemaining,

    required this.isAnswered,

    super.key,
  });

  @override
  State<TimerWidget> createState() => _TimerWidgetState();
}

class _TimerWidgetState extends State<TimerWidget>
    with SingleTickerProviderStateMixin {
  late AnimationController _pulseController;
  late Animation<double> _pulseAnimation;

  @override
  void initState() {
    super.initState();
    _pulseController = AnimationController(
      vsync: this,
      duration: Duration(milliseconds: 800),
    )..repeat(reverse: true);

    _pulseAnimation = Tween<double>(
      begin: 1.0,
      end: 1.1,
    ).animate(_pulseController);
  }

  @override
  void dispose() {
    _pulseController.dispose();

    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    final isWarning = widget.timeRemaining <= 5 && !widget.isAnswered;

    final color = widget.isAnswered
        ? Colors.grey[400]
        : isWarning
        ? Colors.red
        : Colors.deepPurple;

    return Container(
      padding: EdgeInsets.symmetric(horizontal: 16, vertical: 8),
      decoration: BoxDecoration(
        color: color.withOpacity(0.1),
        borderRadius: BorderRadius.circular(8),
        border: Border.all(color: color.withOpacity(0.3)),
      ),
      child: Row(
        mainAxisSize: MainAxisSize.min,
        children: [
          Icon(Icons.timer, color: color, size: 20),
          SizedBox(width: 8),

          if (isWarning && !widget.isAnswered)
            ScaleTransition(
              scale: _pulseAnimation,
              child: Text(
                widget.timeRemaining.toString(),
                style: TextStyle(
                  fontSize: 18,
                  fontWeight: FontWeight.bold,
                  color: color,
                ),
              ),
            )
          else
            Text(
              widget.isAnswered ? 'Answered' : widget.timeRemaining.toString(),
              style: TextStyle(
                fontSize: 18,
                fontWeight: FontWeight.bold,
                color: color,
              ),
            ),
          SizedBox(width: 4),
          Text(
            widget.isAnswered ? '✅' : 'seconds',
            style: TextStyle(fontSize: 12, color: color),
          ),
        ],
      ),
    );
  }
}

      emoji = '👏';
      title = 'Good Job!';

    } else if (percentage >= 40) {
      emoji = '💪';
      title = 'Keep Practicing!';
      color = Colors.orange;
    } else {
      emoji = '📚';
      title = 'Keep Learning!';
      color = Colors.red;
    }

    return Scaffold(
      body: Center(
        child: Padding(
          padding: EdgeInsets.all(24),
          child: FadeTransition(
            opacity: _fadeAnimation,
            child: Column(
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                ScaleTransition(
                  scale: _scaleAnimation,
                  child: Text(emoji, style: TextStyle(fontSize: 72)),
                ),
                SizedBox(height: 16),
                Text(
                  title,
                  style: TextStyle(
                    fontSize: 28,
                    fontWeight: FontWeight.bold,
                    color: color,
                  ),
                ),
                SizedBox(height: 8),
                Text(
                  'You got ${widget.correctAnswers} out of ${widget.totalQuestions} correct!',
                  style: TextStyle(fontSize: 16, color: Colors.grey[600]),
                ),
                SizedBox(height: 32),
                Container(
                  width: 120,
                  height: 120,
                  decoration: BoxDecoration(
                    shape: BoxShape.circle,
                    color: color.withOpacity(0.1),
                    border: Border.all(color: color, width: 4),
                  ),
                  child: Center(
                    child: Column(
                      mainAxisAlignment: MainAxisAlignment.center,
                      children: [
                        Text(
                          widget.score.toString(),
                          style: TextStyle(
                            fontSize: 36,
                            fontWeight: FontWeight.bold,
                            color: color,
                          ),
                        ),
                        Text(
                          'points',
                          style: TextStyle(fontSize: 14, color: color),
                        ),
                      ],
                    ),
                  ),
                ),
                SizedBox(height: 32),
                Row(
                  children: [
                    Expanded(
                      child: _buildStatCard(
                        'Correct',
                        widget.correctAnswers.toString(),
                        Colors.green,
                        Icons.check_circle,
                      ),
                    ),
                    SizedBox(width: 12),
                    Expanded(
                      child: _buildStatCard(
                        'Incorrect',
                        (widget.totalQuestions - widget.correctAnswers)
                            .toString(),
                        Colors.red,
                        Icons.cancel,
                      ),
                    ),
                  ],
                ),
                SizedBox(height: 32),
                SizedBox(
                  width: double.infinity,
                  child: ElevatedButton.icon(
                    onPressed: widget.onRestart,
                    icon: Icon(Icons.refresh),
                    label: Text('Restart Quiz'),
                    style: ElevatedButton.styleFrom(
                      backgroundColor: Colors.deepPurple,
                      foregroundColor: Colors.white,
                      padding: EdgeInsets.symmetric(vertical: 14),
                      shape: RoundedRectangleBorder(
                        borderRadius: BorderRadius.circular(8),
                      ),
                    ),
                  ),
                ),
              ],
            ),
          ),
        ),
      ),
    );
  }

  Widget _buildStatCard(
    String label,
    String value,
    Color color,
    IconData icon,
  ) {
    return Container(
      padding: EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: color.withOpacity(0.05),
        borderRadius: BorderRadius.circular(8),
        border: Border.all(color: color.withOpacity(0.2)),
      ),
      child: Column(
        children: [
          Icon(icon, color: color, size: 24),
          SizedBox(height: 4),
          Text(
            value,
            style: TextStyle(
              fontSize: 20,
              fontWeight: FontWeight.bold,
              color: color,
            ),
          ),
          Text(label, style: TextStyle(fontSize: 12, color: Colors.grey[600])),
        ],
      ),
    );
  }
}

  • State Management – Riverpod with StateNotifier for complex quiz state
  • Page Transitions – Smooth navigation between questions
  • Feedback Animations – Visual feedback for correct/incorrect answers
  • Progress Tracking – Animated progress bar and score display
