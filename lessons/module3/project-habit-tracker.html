Habit Tracker

Build a complete habit tracker app with Hive for storage and notifications for daily reminders and streak tracking.

Hive package
📋 Requirements

What You'll Build

  • A habit tracker app with Hive for persistent storage
  • Create habits – Name, description, frequency (daily/weekly/monthly)
  • Track progress – Mark habits as complete for today
  • Streak tracking – Current and longest streaks
  • Daily reminders – Notifications to complete habits
  • Progress visualization – Weekly/monthly completion stats
  • Habit categories – Organize habits by category
  • Dark mode – Persisted with SharedPreferences

Learning Objectives

  • Implement streak tracking algorithms
  • Use local notifications for reminders
  • Build progress visualization with custom widgets
  • Handle date-based filtering and queries
  • Master Hive with complex relationships

import 'package:flutter_local_notifications/flutter_local_notifications.dart';

import 'package:timezone/timezone.dart' as tz;

import 'package:timezone/data/latest.dart' as tz;

class NotificationService {
  static final FlutterLocalNotificationsPlugin _plugin =
      FlutterLocalNotificationsPlugin();

  static const String channelId = 'habit_channel';

  static const String channelName = 'Habit Reminders';

  static const String channelDescription = 'Reminders for your daily habits';

  static Future<void> initialize() async {
    tz.initializeTimeZones();

    final AndroidInitializationSettings androidSettings =
        AndroidInitializationSettings('@mipmap/ic_launcher');

    final DarwinInitializationSettings iosSettings =
        DarwinInitializationSettings(
          requestAlertPermission: true,
          requestBadgePermission: true,
          requestSoundPermission: true,
        );

    final InitializationSettings settings = InitializationSettings(
      android: androidSettings,
      iOS: iosSettings,
    );

    await _plugin.initialize(settings);

    // Request permissions

    await _plugin
        .resolvePlatformSpecificImplementation<
          AndroidFlutterLocalNotificationsPlugin
        >()
        ?.requestNotificationsPermission();

    await _plugin
        .resolvePlatformSpecificImplementation<
          IOSFlutterLocalNotificationsPlugin
        >()
        ?.requestPermissions(alert: true, badge: true, sound: true);
  }

  static Future<void> showNotification({
    required int id,

    required String title,

    required String body,
  }) async {
    final AndroidNotificationDetails androidDetails =
        AndroidNotificationDetails(
          channelId,
          channelName,
          channelDescription: channelDescription,
          importance: Importance.high,
          priority: Priority.high,
          icon: '@mipmap/ic_launcher',
          playSound: true,
        );

    final DarwinNotificationDetails iosDetails = DarwinNotificationDetails(
      presentAlert: true,
      presentBadge: true,
      presentSound: true,
    );

    final NotificationDetails details = NotificationDetails(
      android: androidDetails,
      iOS: iosDetails,
    );

    await _plugin.show(id, title, body, details);
  }

  static Future<void> scheduleReminder({
    required String habitId,

    required String habitName,

    required String time,
  }) async {
    final timeParts = time.split(':');

    final hour = int.parse(timeParts[0]);

    final minute = int.parse(timeParts[1]);

    final now = DateTime.now();

    final scheduledTime = DateTime(now.year, now.month, now.day, hour, minute);

    final tzLocation = tz.local;

    final scheduledTZ = tz.TZDateTime.from(scheduledTime, tzLocation);

    final id = habitId.hashCode;

    await _plugin.zonedSchedule(
      id,

      '⏰ Habit Reminder',

      'Time to complete: $habitName',
      scheduledTZ,

      const NotificationDetails(
        android: AndroidNotificationDetails(
          channelId,
          channelName,
          channelDescription: channelDescription,
          importance: Importance.high,
          priority: Priority.high,
          icon: '@mipmap/ic_launcher',
        ),
        iOS: DarwinNotificationDetails(),
      ),
      uiLocalNotificationDateInterpretation:
          UILocalNotificationDateInterpretation.absoluteTime,
      matchDateTimeComponents: DateTimeComponents.time,
    );
  }

  static Future<void> cancelReminder(String habitId) async {
    final id = habitId.hashCode;

    await _plugin.cancel(id);
  }

  static Future<void> cancelAllReminders() async {
    await _plugin.cancelAll();
  }

  static Future<void> showDailyReminder() async {
    final today = DateTime.now();

    final dayName = [
      'Monday',

      'Tuesday',

      'Wednesday',

      'Thursday',

      'Friday',

      'Saturday',

      'Sunday',
    ][today.weekday - 1];

    await showNotification(
      id: DateTime.now().millisecondsSinceEpoch.hashCode,
      title: '🌅 Good $dayName!',
      body: 'Don\'t forget to complete your habits today!',
    );
  }
}

class ThemeProvider extends ChangeNotifier {
  bool _isDarkMode = false;

  ThemeProvider() {
    _loadTheme();
  }

  bool get isDarkMode => _isDarkMode;

  Future<void> _loadTheme() async {
    final box = await Hive.openBox('settings');
    _isDarkMode = box.get('dark_mode', defaultValue: false) as bool;
    notifyListeners();
  }

  Future<void> toggleTheme() async {
    _isDarkMode = !_isDarkMode;

    final box = await Hive.openBox('settings');

    await box.put('dark_mode', _isDarkMode);
    notifyListeners();
  }
}

  HabitCategory _selectedCategory = HabitCategory.fitness;

    Colors.blue,

  Future<void> _initBox() async {

  Future<void> _sendDailyReminder() async {
    final box = await Hive.openBox('settings');

    final lastReminderDate = box.get('last_reminder_date') as int?;

    final today = DateTime.now();

    final todayKey = today.year * 10000 + today.month * 100 + today.day;

    if (lastReminderDate == null || lastReminderDate != todayKey) {
      // Check if there are any habits to remind about

      if (_habitBox != null && _habitBox.values.isNotEmpty) {
        await NotificationService.showDailyReminder();

        await box.put('last_reminder_date', todayKey);
      }
    }
  }

  Future<void> _addHabit() async {
    final name = _nameController.text.trim();

    final description = _descController.text.trim();

    if (name.isEmpty) return;

    final habit = Habit(
      id: DateTime.now().millisecondsSinceEpoch.toString(),
      name: name,
      description: description.isNotEmpty ? description : 'No description',
      frequency: _selectedFrequency,
      category: _selectedCategory,
      reminderTime: _reminderTime,
    );

    await _habitBox.put(habit.id, habit);

    // Schedule reminder if set

    if (_reminderTime != null) {
      await NotificationService.scheduleReminder(
        habitId: habit.id,
        habitName: habit.name,
        time: _reminderTime!,
      );
    }

    _nameController.clear();
    _descController.clear();
    _reminderTime = null;
    setState(() {});
  }

  Future<void> _toggleHabitCompletion(String id) async {
    final habit = _habitBox.get(id);

    if (habit == null) return;

    final isCompleted = habit.isCompletedToday();
    List<DateTime> newCompletedDates = List.from(habit.completedDates);

    if (isCompleted) {
      // Remove today's completion

      newCompletedDates.removeWhere(
        (date) =>
            date.year == DateTime.now().year &&
            date.month == DateTime.now().month &&
            date.day == DateTime.now().day,
      );
    } else {
      // Add today's completion

      newCompletedDates.add(DateTime.now());
    }

    // Recalculate streaks

    final streaks = _calculateStreaks(newCompletedDates);

    final updatedHabit = habit.copyWith(
      completedDates: newCompletedDates,
      currentStreak: streaks.current,
      longestStreak: streaks.longest,
    );

    await _habitBox.put(id, updatedHabit);

    // Show feedback notification

    if (!isCompleted && habit.reminderTime != null) {
      if (updatedHabit.currentStreak % 7 == 0 &&
          updatedHabit.currentStreak > 0) {
        await NotificationService.showNotification(
          id: DateTime.now().millisecondsSinceEpoch.hashCode,
          title: '🎉 Streak Milestone!',
          body:
              'You\'ve completed "$habitName" for ${updatedHabit.currentStreak} days in a row!',
        );
      }
    }

    setState(() {});
  }

  Future<void> _deleteHabit(String id) async {
    final habit = _habitBox.get(id);

    if (habit == null) return;

    showDialog(
      context: context,
      builder: (context) => AlertDialog(
        title: Text('Delete Habit'),
        content: Text(
          'Are you sure you want to delete "$habit.name"? This cannot be undone.',
        ),
        actions: [
          TextButton(
            onPressed: () => Navigator.pop(context),
            child: Text('Cancel'),
          ),
          ElevatedButton(
            onPressed: () async {
              // Cancel reminder

              if (habit.reminderTime != null) {
                await NotificationService.cancelReminder(habit.id);
              }

              await _habitBox.delete(id);
              Navigator.pop(context);
              setState(() {});
            },
            child: Text('Delete'),
            style: ElevatedButton.styleFrom(backgroundColor: Colors.red),
          ),
        ],
      ),
    );
  }

  Future<void> _archiveHabit(String id) async {
    final habit = _habitBox.get(id);

    if (habit == null) return;

    final updated = habit.copyWith(isArchived: !habit.isArchived);

    await _habitBox.put(id, updated);

    // Cancel reminder if archived

    if (updated.isArchived && habit.reminderTime != null) {
      await NotificationService.cancelReminder(habit.id);
    }

    setState(() {});
  }

  StreakResult _calculateStreaks(List<DateTime> completedDates) {
    if (completedDates.isEmpty) {
      return StreakResult(current: 0, longest: 0);
    }

    final sortedDates = List<DateTime>.from(completedDates)
      ..sort((a, b) => b.compareTo(a));

    // Calculate current streak

    int currentStreak = 0;

    var checkDate = DateTime.now();

    for (var date in sortedDates) {
      if (date.year == checkDate.year &&
          date.month == checkDate.month &&
          date.day == checkDate.day) {
        currentStreak++;
        checkDate = checkDate.subtract(Duration(days: 1));
      } else {
        break;
      }
    }

    // Calculate longest streak

    int longestStreak = 0;

    int tempStreak = 0;

    var prevDate = DateTime.now();

    for (var i = 0; i < sortedDates.length; i++) {
      final date = sortedDates[i];

      if (i == 0) {
        tempStreak = 1;
        prevDate = date;
      } else {
        final difference = prevDate.difference(date).inDays;

        if (difference == 1) {
          tempStreak++;
        } else if (difference > 1) {
          tempStreak = 1;
        }
        prevDate = date;
      }

      if (tempStreak > longestStreak) {
        longestStreak = tempStreak;
      }
    }

    return StreakResult(current: currentStreak, longest: longestStreak);
  }

  Color _getStreakColor(int streak) {
    final index = streak.clamp(0, _streakColors.length - 1);

    return _streakColors[index];

// 6. PROVIDER HELPERS

class Provider<T extends ChangeNotifier> extends InheritedWidget {
  final T notifier;

  Provider({required this.notifier, required Widget child})
    : super(child: child);

  static T of<T extends ChangeNotifier>(BuildContext context) {
    return context.dependOnInheritedWidgetOfExactType<Provider<T>>()!.notifier;
  }

  @override
  bool updateShouldNotify(Provider oldWidget) => notifier != oldWidget.notifier;
}

class ChangeNotifierProvider<T extends ChangeNotifier> extends StatefulWidget {
  final T Function() create;

  final Widget child;

  const ChangeNotifierProvider({required this.create, required this.child});
  @override
  State<ChangeNotifierProvider<T>> createState() =>
      _ChangeNotifierProviderState<T>();
}

class _ChangeNotifierProviderState<T extends ChangeNotifier>
    extends State<ChangeNotifierProvider<T>> {
  late T notifier;
  @override
  void initState() {
    super.initState();
    notifier = widget.create();
  }

  @override
  Widget build(BuildContext context) =>
      Provider<T>(notifier: notifier, child: widget.child);
}

class Consumer<T extends ChangeNotifier> extends StatelessWidget {
  final Widget Function(BuildContext context, T notifier, Widget? child)
  builder;

  final Widget? child;

  const Consumer({required this.builder, this.child});
  @override
  Widget build(BuildContext context) {
    final notifier = Provider.of<T>(context);

    return AnimatedBuilder(
      animation: notifier,
      builder: (context, _) => builder(context, notifier, child),
    );
  }
}
