Netflix Style UI

Build a stunning Netflix-inspired streaming app interface with slivers, custom scroll views, and hero animations.

Project Overview

In this project, you'll build a Netflix Style UI — a stunning streaming app interface with a collapsing header, horizontal scrolling rows, and smooth hero transitions. You'll master slivers , custom scroll views , and hero animations to create a polished, professional streaming experience.

📚 Prerequisites

  • Understanding of Slivers
  • Understanding of Hero Animations
  • Understanding of Custom Widgets
  • Familiarity with Animations

🎯 Learning Objectives

  • Master slivers for complex scrolling layouts
  • Build collapsing headers with SliverAppBar
  • Create horizontal scrolling content rows
  • Implement hero animations for content transitions
  • Build a Netflix-style UI with custom scroll views
  • Combine multiple sliver types for a polished experience

class Movie {
  final String id;

  final String title;

  final String description;

  final String imageUrl;

  final String backdropUrl;

    required this.backdropUrl,

    required this.genre,

    required this.year,

    required this.rating,

    required this.duration,

}

Sliver Components

1. Content Row (SliverList)

lib/widgets/content_row.dart 
class ContentRow extends StatelessWidget {
  final String title;

  final List<Movie> movies;

  const ContentRow({required this.title, required this.movies, super.key});

  @override
  Widget build(BuildContext context) {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Padding(
          padding: EdgeInsets.symmetric(horizontal: 16, vertical: 8),
          child: Text(
            title,
            style: TextStyle(
              fontSize: 20,
              fontWeight: FontWeight.bold,
              color: Colors.white,
            ),
          ),
        ),
        SizedBox(
          height: 220,
          child: ListView.builder(
            scrollDirection: Axis.horizontal,
            padding: EdgeInsets.symmetric(horizontal: 8),
            itemCount: movies.length,
            itemBuilder: (context, index) {
              return Padding(
                padding: EdgeInsets.symmetric(horizontal: 8),
                child: MovieCard(movie: movies[index]),
              );
            },
          ),
        ),
      ],
    );
  }
}

2. Movie Card with Hero

lib/widgets/movie_card.dart 
class MovieCard extends StatelessWidget {
  final Movie movie;

  final bool isLarge;

  const MovieCard({required this.movie, this.isLarge = false, super.key});

  double get cardWidth => isLarge ? 200 : 140;

  double get cardHeight => isLarge ? 280 : 200;

  @override
  Widget build(BuildContext context) {
    return GestureDetector(
      onTap: () {
        Navigator.push(
          context,
          MaterialPageRoute(
            builder: (context) => MovieDetailScreen(movie: movie),
          ),
        );
      },
      child: Hero(
        tag: 'movie_${movie.id}',
        child: ClipRRect(
          borderRadius: BorderRadius.circular(8),
          child: Container(
            width: cardWidth,
            height: cardHeight,
            decoration: BoxDecoration(
              image: DecorationImage(
                image: NetworkImage(movie.imageUrl),
                fit: BoxFit.cover,
              ),
            ),
            child: Container(
              decoration: BoxDecoration(
                gradient: LinearGradient(
                  begin: Alignment.bottomCenter,
                  end: Alignment.topCenter,
                  colors: [Colors.black54, Colors.transparent],
                ),
              ),
              child: Align(
                alignment: Alignment.bottomLeft,
                child: Padding(
                  padding: EdgeInsets.all(8),
                  child: Text(
                    movie.title,
                    style: TextStyle(
                      color: Colors.white,
                      fontSize: 12,
                      fontWeight: FontWeight.bold,
                    ),
                    maxLines: 2,
                    overflow: TextOverflow.ellipsis,
                  ),
                ),
              ),
            ),
          ),
        ),
      ),
    );
  }
}

3. Hero Image for Detail Screen

lib/widgets/hero_image.dart 
class HeroImage extends StatelessWidget {
  final String imageUrl;

  final String tag;

  final double height;

  const HeroImage({
    required this.imageUrl,

    required this.tag,

    this.height = 300,

    super.key,
  });

  @override
  Widget build(BuildContext context) {
    return Hero(
      tag: tag,
      child: Container(
        height: height,
        width: double.infinity,
        decoration: BoxDecoration(
          image: DecorationImage(
            image: NetworkImage(imageUrl),
            fit: BoxFit.cover,
          ),
        ),
        child: Container(
          decoration: BoxDecoration(
            gradient: LinearGradient(
              begin: Alignment.bottomCenter,
              end: Alignment.topCenter,
              colors: [Colors.black87, Colors.transparent],
              stops: [0.0, 0.4],
            ),
          ),
        ),
      ),
    );
  }
}
Home Screen

The Home Screen uses a CustomScrollView with multiple sliver types:

lib/screens/home_screen.dart 
class HomeScreen extends StatelessWidget {
  final List<ContentRow> rows = MovieData.rows;

  final Movie featured = MovieData.trending.first;

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: Colors.black,
      body: CustomScrollView(
        slivers: [
          // Collapsible App Bar
          SliverAppBar(
            expandedHeight: 500,
            collapsedHeight: 80,
            floating: true,
            pinned: true,
            backgroundColor: Colors.black,
            flexibleSpace: FlexibleSpaceBar(
              title: Row(
                children: [
                  Icon(Icons.play_circle_filled, color: Colors.red, size: 28),
                  SizedBox(width: 4),
                  Text(
                    'NETFLIX',
                    style: TextStyle(
                      fontSize: 20,
                      fontWeight: FontWeight.bold,
                      color: Colors.red,
                      letterSpacing: 2,
                    ),
                  ),
                ],
              ),
              centerTitle: false,
              titlePadding: EdgeInsets.only(left: 16, bottom: 16),
              background: Stack(
                fit: StackFit.expand,
                children: [
                  HeroImage(
                    imageUrl: featured.backdropUrl,
                    tag: 'movie_${featured.id}',
                    height: 500,
                  ),

                  // Featured content overlay
                  Positioned(
                    bottom: 80,
                    left: 16,
                    right: 16,
                    child: Column(
                      crossAxisAlignment: CrossAxisAlignment.start,
                      children: [
                        Text(
                          featured.title,
                          style: TextStyle(
                            color: Colors.white,
                            fontSize: 32,
                            fontWeight: FontWeight.bold,
                          ),
                        ),
                        SizedBox(height: 8),
                        Row(
                          children: [
                            _buildBadge('${featured.year}'),
                            _buildBadge('${featured.duration} min'),
                            _buildBadge('${featured.rating} ⭐'),
                          ],
                        ),
                        SizedBox(height: 12),
                        Row(
                          children: [
                            _buildActionButton(
                              Icons.play_arrow,

                              'Play',
                              Colors.white,
                            ),
                            SizedBox(width: 12),
                            _buildActionButton(
                              Icons.add,

                              'My List',
                              Colors.grey[700]!,
                            ),
                          ],
                        ),
                      ],
                    ),
                  ),
                ],
              ),
            ),
          ),

          // Content rows using SliverList
          ...rows.map(
            (row) => SliverToBoxAdapter(
              child: ContentRow(title: row.title, movies: row.movies),
            ),
          ),
        ],
      ),

      // Bottom Navigation
      bottomNavigationBar: Container(
        color: Colors.black,
        child: Padding(
          padding: EdgeInsets.symmetric(vertical: 8),
          child: Row(
            mainAxisAlignment: MainAxisAlignment.spaceEvenly,
            children: [
              _buildNavItem(Icons.home, 'Home', true),
              _buildNavItem(Icons.search, 'Search', false),
              _buildNavItem(Icons.video_library, 'My List', false),
              _buildNavItem(Icons.person, 'Profile', false),
            ],
          ),
        ),
      ),
    );
  }

  Widget _buildBadge(String text) {
    return Container(
      padding: EdgeInsets.symmetric(horizontal: 8, vertical: 2),
      margin: EdgeInsets.only(right: 8),
      decoration: BoxDecoration(
        border: Border.all(color: Colors.white54),
        borderRadius: BorderRadius.circular(4),
      ),
      child: Text(text, style: TextStyle(color: Colors.white70, fontSize: 12)),
    );
  }

  Widget _buildActionButton(IconData icon, String label, Color bgColor) {
    return ElevatedButton.icon(
      onPressed: () {},
      icon: Icon(
        icon,
        color: bgColor == Colors.white ? Colors.black : Colors.white,
      ),
      label: Text(
        label,
        style: TextStyle(
          color: bgColor == Colors.white ? Colors.black : Colors.white,
          fontWeight: FontWeight.bold,
        ),
      ),
      style: ElevatedButton.styleFrom(
        backgroundColor: bgColor,
        foregroundColor: Colors.black,
        padding: EdgeInsets.symmetric(horizontal: 24, vertical: 10),
        shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(4)),
      ),
    );
  }

  Widget _buildNavItem(IconData icon, String label, bool isActive) {
    return Column(
      mainAxisSize: MainAxisSize.min,
      children: [
        Icon(icon, color: isActive ? Colors.white : Colors.grey[600]),
        Text(
          label,
          style: TextStyle(
            color: isActive ? Colors.white : Colors.grey[600],
            fontSize: 10,
          ),
        ),
      ],
    );
  }
}
Movie Detail Screen

The Detail Screen uses a hero animation for the backdrop image:

lib/screens/movie_detail_screen.dart 
class MovieDetailScreen extends StatelessWidget {
  final Movie movie;

  const MovieDetailScreen({required this.movie, super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: Colors.black,
      body: CustomScrollView(
        slivers: [
          // Hero image with SliverAppBar
          SliverAppBar(
            expandedHeight: 400,
            floating: false,
            pinned: true,
            backgroundColor: Colors.black,
            flexibleSpace: FlexibleSpaceBar(
              background: HeroImage(
                imageUrl: movie.backdropUrl,
                tag: 'movie_${movie.id}',
                height: 400,
              ),
            ),
            leading: IconButton(
              onPressed: () => Navigator.pop(context),
              icon: Icon(Icons.arrow_back, color: Colors.white),
            ),
          ),

          // Movie details
          SliverToBoxAdapter(
            child: Padding(
              padding: EdgeInsets.all(24),
              child: Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                children: [
                  // Title and rating
                  Row(
                    mainAxisAlignment: MainAxisAlignment.spaceBetween,
                    children: [
                      Expanded(
                        child: Text(
                          movie.title,
                          style: TextStyle(
                            color: Colors.white,
                            fontSize: 28,
                            fontWeight: FontWeight.bold,
                          ),
                        ),
                      ),
                      Container(
                        padding: EdgeInsets.symmetric(
                          horizontal: 12,
                          vertical: 6,
                        ),
                        decoration: BoxDecoration(
                          color: Colors.green,
                          borderRadius: BorderRadius.circular(4),
                        ),
                        child: Text(
                          movie.rating.toStringAsFixed(1),
                          style: TextStyle(
                            color: Colors.white,
                            fontWeight: FontWeight.bold,
                          ),
                        ),
                      ),
                    ],
                  ),
                  SizedBox(height: 8),
                  Text(
                    movie.genre,
                    style: TextStyle(color: Colors.grey[400], fontSize: 16),
                  ),
                  SizedBox(height: 8),
                  Row(
                    children: [
                      Text(
                        '${movie.year}',
                        style: TextStyle(color: Colors.grey[400]),
                      ),
                      Text(' • ', style: TextStyle(color: Colors.grey[400])),
                      Text(
                        '${movie.duration} min',
                        style: TextStyle(color: Colors.grey[400]),
                      ),
                    ],
                  ),
                  SizedBox(height: 16),
                  Text(
                    movie.description,
                    style: TextStyle(
                      color: Colors.white70,
                      fontSize: 16,
                      height: 1.5,
                    ),
                  ),
                  SizedBox(height: 24),

                  // Action buttons
                  Row(
                    children: [
                      Expanded(
                        child: ElevatedButton.icon(
                          onPressed: () {},
                          icon: Icon(Icons.play_arrow, color: Colors.black),
                          label: Text(
                            'Play',
                            style: TextStyle(
                              color: Colors.black,
                              fontWeight: FontWeight.bold,
                            ),
                          ),
                          style: ElevatedButton.styleFrom(
                            backgroundColor: Colors.white,
                            padding: EdgeInsets.symmetric(vertical: 14),
                            shape: RoundedRectangleBorder(
                              borderRadius: BorderRadius.circular(4),
                            ),
                          ),
                        ),
                      ),
                      SizedBox(width: 12),
                      Expanded(
                        child: ElevatedButton.icon(
                          onPressed: () {},
                          icon: Icon(Icons.add, color: Colors.white),
                          label: Text(
                            'My List',
                            style: TextStyle(
                              color: Colors.white,
                              fontWeight: FontWeight.bold,
                            ),
                          ),
                          style: ElevatedButton.styleFrom(
                            backgroundColor: Colors.grey[800],
                            padding: EdgeInsets.symmetric(vertical: 14),
                            shape: RoundedRectangleBorder(
                              borderRadius: BorderRadius.circular(4),
                            ),
                          ),
                        ),
                      ),
                    ],
                  ),
                  SizedBox(height: 24),

                  // Director
                  Text(
                    'Director',
                    style: TextStyle(color: Colors.grey[400], fontSize: 14),
                  ),
                  Text(
                    movie.director,
                    style: TextStyle(
                      color: Colors.white,
                      fontSize: 16,
                      fontWeight: FontWeight.w500,
                    ),
                  ),
                  SizedBox(height: 16),

                  // Cast
                  Text(
                    'Cast',
                    style: TextStyle(color: Colors.grey[400], fontSize: 14),
                  ),
                  Wrap(
                    spacing: 8,
                    runSpacing: 4,
                    children: movie.cast
                        .map(
                          (actor) => Chip(
                            label: Text(
                              actor,
                              style: TextStyle(color: Colors.white),
                            ),
                            backgroundColor: Colors.grey[800],
                          ),
                        )
                        .toList(),
                  ),
                ],
              ),
            ),
          ),
        ],
      ),
    );
  }
