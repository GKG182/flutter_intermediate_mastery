E-Commerce UI

Project Overview

In this project, you'll build a responsive E-Commerce UI — a modern, state-driven shopping app with product browsing, filtering, shopping cart management, and a responsive layout that adapts to any screen size.

📚 Prerequisites

  • Understanding of Riverpod Basics
  • Understanding of Responsive Design
  • Understanding of Custom Widgets
  • Familiarity with Adaptive Design

🎯 Learning Objectives

  • Build a responsive e-commerce UI with adaptive layouts
  • Implement state management with Riverpod
  • Create a shopping cart system with add/remove functionality
  • Build product filtering and search
  • Handle different screen sizes (mobile, tablet, desktop)
  • Create reusable UI components

lib/models/product.dart class Product { final String id; final String name; final String description; final double price; final String imageUrl; final String category; final double rating; final int reviewCount; final bool inStock; Product ({ required this .id, required this .name, required this .description, required this .price, required this .imageUrl, required this .category, this .rating = 0.0 , this .reviewCount = 0 , this .inStock = true , }); } class CartItem { final Product product; int quantity; CartItem ({ required this .product, this .quantity = 1 , }); double get totalPrice => product.price * quantity; }
lib/providers/cart_provider.dart import 'package:flutter_riverpod/flutter_riverpod.dart' ; import '../models/product.dart' ; class CartNotifier extends StateNotifier<List<CartItem>> { CartNotifier () : super ([]); void addItem (Product product, { int quantity = 1 }) { final existingIndex = state. indexWhere ( (item) => item.product.id == product.id, ); if (existingIndex != - 1 ) { // Product already in cart, update quantity final updatedItem = state[existingIndex]; state = [ ...state. sublist ( 0 , existingIndex), CartItem( product: updatedItem.product, quantity: updatedItem.quantity + quantity, ), ...state. sublist (existingIndex + 1 ), ]; } else { // New product, add to cart state = [...state, CartItem(product: product, quantity: quantity)]; } } void removeItem (String productId) { state = state. where ((item) => item.product.id != productId). toList (); } void updateQuantity (String productId, int quantity) { final index = state. indexWhere ((item) => item.product.id == productId); if (index != - 1 ) { if (quantity <= 0 ) { removeItem (productId); } else { state = [ ...state. sublist ( 0 , index), CartItem( product: state[index].product, quantity: quantity, ), ...state. sublist (index + 1 ), ]; } } } void clearCart () { state = []; } int get totalItems => state. fold ( 0 , (sum, item) => sum + item.quantity); double get totalPrice => state. fold ( 0.0 , (sum, item) => sum + item.totalPrice); } final cartProvider = StateNotifierProvider<CartNotifier, List<CartItem>>((ref) { return CartNotifier(); }); final cartTotalItemsProvider = Provider< int >((ref) { return ref. watch (cartProvider). fold ( 0 , (sum, item) => sum + item.quantity); }); final cartTotalPriceProvider = Provider< double >((ref) { return ref. watch (cartProvider). fold ( 0.0 , (sum, item) => sum + item.totalPrice); });
lib/providers/product_provider.dart import 'package:flutter_riverpod/flutter_riverpod.dart' ; import '../models/product.dart' ; import '../data/products_data.dart' ; final productsProvider = Provider<List<Product>>((ref) { return ProductData.products; }); final productByIdProvider = Provider<Product Function(String)>((ref) { final products = ref. watch (productsProvider); return (String id) { return products. firstWhere ((p) => p.id == id); }; }); final filteredProductsProvider = Provider<List<Product>>((ref) { final products = ref. watch (productsProvider); final category = ref. watch (categoryFilterProvider); final searchQuery = ref. watch (searchQueryProvider); return products. where ((product) { final matchesCategory = category == 'All' || product.category == category; final matchesSearch = product.name. toLowerCase () . contains (searchQuery. toLowerCase ()); return matchesCategory && matchesSearch; }). toList (); }); final categoryFilterProvider = StateProvider<String>((ref) { return 'All' ; }); final searchQueryProvider = StateProvider<String>((ref) { return '' ; });
Responsive Layout

With Riverpod managing the state, the cart updates in real-time across the app. This project demonstrates how to combine responsive design, state management, and modern UI patterns to create a professional e-commerce experience.
